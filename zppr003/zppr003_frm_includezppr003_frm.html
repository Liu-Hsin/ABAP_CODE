*&      Form  FRM_DOWNLOAD_EXCEL
*&---------------------------------------------------------------------*
*       下载文件模板
*----------------------------------------------------------------------*
FORM frm_download_excel USING p_file TYPE c.
  DATA: lf_string   TYPE string,
        ls_objdata  LIKE wwwdatatab,
        lf_rc       LIKE sy-subrc,
        application TYPE ole2_object,
        workbook    TYPE ole2_object,
        sheet       TYPE ole2_object.
*
  IF sy-ucomm = c_fc01.
    SELECT SINGLE relid
                  objid
              INTO CORRESPONDING FIELDS OF ls_objdata
              FROM wwwdata
             WHERE srtf2 = 0
               AND relid = 'MI'
               AND objid = c_objid.
    ENDIF.
*
    IF sy-subrc <> 0.
      MESSAGE c_text1 TYPE 'E'.
      RETURN.
    ENDIF.

    IF ls_objdata-objid = space .
      MESSAGE c_text1 TYPE 'E'.
      RETURN.
    ENDIF.

    CONDENSE p_file.

    CALL FUNCTION 'DOWNLOAD_WEB_OBJECT'
      EXPORTING
        key         = ls_objdata
        destination = p_file
      IMPORTING
        rc          = lf_rc.
    IF lf_rc <> 0.
      MESSAGE c_text2 TYPE 'E'.
      RETURN.
    ENDIF.

* 打开模版 文件
    CREATE OBJECT application 'EXCEL.APPLICATION'.
    IF sy-subrc NE 0.
      MESSAGE e000 WITH c_text3.
    ENDIF.

    CALL METHOD OF
      application
        'WORKBOOKS' = workbook.

    CALL METHOD OF
      workbook
      'OPEN'
      EXPORTING
        #1 = p_file.

    CALL METHOD OF
        application
        'WORKSHEETS' = sheet
      EXPORTING
        #1           = 1.

    GET PROPERTY OF  application 'ACTIVESHEET' = sheet.
    SET PROPERTY OF application 'VISIBLE' = 1.

    FREE sheet.
    FREE workbook.
    FREE application.
    CALL METHOD OF
      workbook
      'CLOSE'.
    CALL METHOD OF
      application
      'QUIT'.
ENDFORM.
