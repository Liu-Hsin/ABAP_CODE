FORM set_config_values.

  DATA: l_date_long     TYPE c LENGTH 10,
        l_time_long     TYPE c LENGTH 8,
        ls_util_detail  TYPE srt_util_detail.

  ls_util_detail = cl_soap_util_db=>admin_get_entry(
                     utilid = m_select_utilid
                     idtype = m_select_idtype
                   ).

  IF ls_util_detail-perf_level IS NOT INITIAL OR
     ls_util_detail-func_level IS NOT INITIAL OR
     ls_util_detail-pload_level IS NOT INITIAL.
    WRITE: ls_util_detail-expiry_date TO l_date_long,
           ls_util_detail-expiry_time TO l_time_long.
    CONCATENATE l_date_long l_time_long
      INTO m_cfg_validation SEPARATED BY space.
  ELSE.
    CLEAR m_cfg_validation.
  ENDIF.

  data lv_expiry_ts type timestamp.
  get time stamp field lv_expiry_ts.
  lv_expiry_ts = cl_abap_tstmp=>add( tstmp = lv_expiry_ts secs  = cl_soap_util_db=>co_trace_validation ).
  if lv_expiry_ts < ls_util_detail-expiry_ts.
    m_cfg_expiry = ls_util_detail-expiry_ts.
  else.
    m_cfg_expiry = lv_expiry_ts.
  endif.

  m_last_user = ls_util_detail-last_user.

  IF ls_util_detail-pload_side O if_soap_util_constants=>pload_side_consider_bl.
    cb_consider_bl = 'X'.
  else.
    cb_consider_bl = space.
  endif.

  IF ls_util_detail IS NOT INITIAL.
    srt_util_config-perf_level  = ls_util_detail-perf_level.
    srt_util_config-func_level  = ls_util_detail-func_level.
    srt_util_config-pload_level = ls_util_detail-pload_level.
    srt_util_config-cpro_1      = ls_util_detail-cpro_1.
    srt_util_config-cpro_2      = ls_util_detail-cpro_2.
    srt_util_config-cpro_3      = ls_util_detail-cpro_3.
    srt_util_config-sdef_1      = ls_util_detail-sdef_1.
    srt_util_config-sdef_2      = ls_util_detail-sdef_2.
    srt_util_config-sdef_3      = ls_util_detail-sdef_3.
    IF ls_util_detail-pload_side O if_soap_util_constants=>pload_side_cons AND
       ls_util_detail-pload_side O if_soap_util_constants=>pload_side_prov.
      rb_side_both = 'X'.
      CLEAR: rb_side_cons, rb_side_prov.
    ELSE.
      IF ls_util_detail-pload_side O if_soap_util_constants=>pload_side_cons.
        rb_side_cons = 'X'.
        CLEAR: rb_side_prov, rb_side_both.
      ENDIF.
      IF ls_util_detail-pload_side O if_soap_util_constants=>pload_side_prov.
        rb_side_prov = 'X'.
        CLEAR: rb_side_cons, rb_side_both.
      ENDIF.
    ENDIF.
    IF ls_util_detail-pload_side O if_soap_util_constants=>pload_side_cpro_spe.
      rb_cpro_spe = 'X'.
      CLEAR rb_cpro_any.
    ELSE.
      rb_cpro_any = 'X'.
      CLEAR rb_cpro_spe.
    ENDIF.
    IF ls_util_detail-pload_side O if_soap_util_constants=>pload_side_sdef_spe.
      rb_sdef_spe = 'X'.
      CLEAR rb_sdef_any.
    ELSE.
      rb_sdef_any = 'X'.
      CLEAR rb_sdef_spe.
    ENDIF.
  ELSE.
    ls_util_detail-utilid = m_select_utilid.
    ls_util_detail-idtype = m_select_idtype.
    IF m_select_utilid IS NOT INITIAL.
      m_error_message = cl_soap_util_db=>admin_create(
                          utilid = m_select_utilid
                          idtype = m_select_idtype
                        ).
      IF m_error_message IS NOT INITIAL.
        MESSAGE i208(00) WITH m_error_message.
        EXIT.
      ENDIF.
    ENDIF.
    CLEAR: srt_util_config-perf_level, srt_util_config-func_level, srt_util_config-pload_level,
           srt_util_config-cpro_1, srt_util_config-cpro_2, srt_util_config-cpro_3,
           srt_util_config-sdef_1, srt_util_config-sdef_2, srt_util_config-sdef_3.
  ENDIF.

ENDFORM. " set_config_values
