FORM frm_get_data_01.

  DATA: ls_acdoct TYPE ts_acdoct,
        lt_acdoct TYPE TABLE OF ts_acdoct.
  DATA: lv_no(2)                  TYPE n,

        lv_lastmonth_acc          TYPE hslxx12,
        lv_lastyear_dmbtr_acc     TYPE hslxx12,
        lv_year_dmbtr_add_acc     TYPE hslxx12,
        lv_lastyear_dmbtr_add_acc TYPE hslxx12.
  DATA: lv_year                    TYPE bseg-gjahr,
        lv_lastyear                TYPE bseg-gjahr,
        lv_year_dmbtr_add_year     TYPE bseg-gjahr,
        lv_lastyear_dmbtr_add_year TYPE bseg-gjahr,
        lv_month                   TYPE bseg-h_monat,
        lv_lastmonth               TYPE bseg-h_monat,
        lv_date                    TYPE sy-datum.
  FIELD-SYMBOLS <fs>.

  lv_date = |{ p_gjahr }{ p_monat }{ '01' }|.
  PERFORM frm_get_last_month USING 1 lv_date.
  lv_year = lv_date+0(4).
  lv_month = lv_date+4(2).
  PERFORM frm_get_last_month USING 1 lv_date.
  lv_lastyear = lv_year - 1.
  lv_lastmonth = lv_date+4(2).
  lv_year_dmbtr_add_year = p_gjahr - 1.
  lv_lastyear_dmbtr_add_year = lv_year_dmbtr_add_year - 1.

  SELECT SUM( hsl01 ) AS hsl01, SUM( hsl02 ) AS hsl02, SUM( hsl03 ) AS hsl03, SUM( hsl04 ) AS hsl04,
  SUM( hsl05 ) AS hsl05, SUM( hsl06 ) AS hsl06, SUM( hsl07 ) AS hsl07, SUM( hsl08 ) AS hsl08,
  SUM( hsl09 ) AS hsl09, SUM( hsl10 ) AS hsl10, SUM( hsl11 ) AS hsl11, SUM( hsl12 ) AS hsl12,
  SUM( hslvt ) AS hslvt, ryear
  FROM v_acdoct_view
  WHERE rbukrs = @p_bukrs
  AND ( ryear = @lv_year OR ryear = @lv_lastyear OR ryear = @lv_lastyear_dmbtr_add_year )
  AND ( left( racct, 4 ) = '1001' OR left( racct, 4 ) = '1002' OR left( racct, 4 ) = '1012' )
  GROUP BY ryear
  ORDER BY ryear
  INTO CORRESPONDING FIELDS OF TABLE @lt_acdoct.

  READ TABLE gt_standard INTO gs_standard INDEX TEXT-038.

  LOOP AT lt_acdoct INTO ls_acdoct.

    IF ls_acdoct-ryear = lv_year.
      "计算本月数和上月数"
      lv_no = '00'.
      DO lv_month TIMES.
        lv_no = lv_no + 1.
        CONCATENATE 'hsl' lv_no INTO lv_field.
        CONDENSE lv_field.
        ASSIGN COMPONENT lv_field OF STRUCTURE ls_acdoct TO <fs>.
        lv_month_acc = lv_month_acc + <fs>.
        IF sy-index <> lv_month.
          lv_lastmonth_acc = lv_lastmonth_acc + <fs>.
        ENDIF.
      ENDDO.
      lv_month_acc = lv_month_acc + ls_acdoct-hslvt.
      lv_lastmonth_acc = lv_lastmonth_acc + ls_acdoct-hslvt.
    ENDIF.
    IF ls_acdoct-ryear = lv_lastyear.
      "计算上年同期数
      lv_no = '00'.
      DO lv_month TIMES.
        lv_no = lv_no + 1.
        CONCATENATE 'hsl' lv_no INTO lv_field.
        CONDENSE lv_field.
        ASSIGN COMPONENT lv_field OF STRUCTURE ls_acdoct TO <fs>.
        lv_lastyear_dmbtr_acc = lv_lastyear_dmbtr_acc + <fs>.
      ENDDO.
      lv_lastyear_dmbtr_acc = lv_lastyear_dmbtr_acc + ls_acdoct-hslvt.
    ENDIF.
    IF ls_acdoct-ryear = lv_year_dmbtr_add_year.
      "计算本年累计
      lv_year_dmbtr_add_no = '00'.
      DO 12 TIMES.
        lv_year_dmbtr_add_no = lv_year_dmbtr_add_no + 1.
        CONCATENATE 'hsl' lv_year_dmbtr_add_no INTO lv_field.
        CONDENSE lv_field.
        ASSIGN COMPONENT lv_field OF STRUCTURE ls_acdoct TO <fs>.
        lv_year_dmbtr_add_acc = lv_year_dmbtr_add_acc + <fs>.
      ENDDO.
      lv_year_dmbtr_add_acc = lv_year_dmbtr_add_acc + ls_acdoct-hslvt.
    ENDIF.
    IF ls_acdoct-ryear = lv_lastyear_dmbtr_add_year.
      "计算去年累计
      lv_year_dmbtr_add_no = '00'.
      DO 12 TIMES.
        lv_year_dmbtr_add_no = lv_year_dmbtr_add_no + 1.
        CONCATENATE 'hsl' lv_year_dmbtr_add_no INTO lv_field.
        CONDENSE lv_field.
        ASSIGN COMPONENT lv_field OF STRUCTURE ls_acdoct TO <fs>.
        lv_lastyear_dmbtr_add_acc = lv_lastyear_dmbtr_add_acc + <fs>.
      ENDDO.
      lv_lastyear_dmbtr_add_acc = lv_lastyear_dmbtr_add_acc + ls_acdoct-hslvt.
    ENDIF.
  ENDLOOP.

  gs_standard-month_dmbtr = lv_month_acc.
  gs_standard-lastmonth_dmbtr = lv_lastmonth_acc.
  gs_standard-lastyear_dmbtr = lv_lastyear_dmbtr_acc.
  gs_standard-year_dmbtr_add = lv_year_dmbtr_add_acc.
  gs_standard-lastyear_dmbtr_add = lv_lastyear_dmbtr_add_acc.
  MODIFY gt_standard FROM gs_standard TRANSPORTING month_dmbtr lastmonth_dmbtr lastyear_dmbtr year_dmbtr_add lastyear_dmbtr_add
  WHERE line_no = gs_standard-line_no.
ENDFORM.

*& Form frm_calculate_month_dmbtr
*&---------------------------------------------------------------------*
*& text 计算本月数、上月数、上年同期数
*&---------------------------------------------------------------------*
FORM frm_calculate_month_dmbtr USING ls_temp TYPE ts_standard VALUE(monat) VALUE(gjahr) ls_month_flag.
  IF monat = ls_temp-h_monat AND gjahr = ls_temp-gjahr.
    CLEAR gs_standard.
    READ TABLE gt_standard INTO gs_standard WITH KEY rstgr = ls_temp-rstgr.
    CASE ls_month_flag.
      WHEN 'month_dmbtr'.
        gs_standard-month_dmbtr = ls_temp-month_dmbtr + gs_standard-month_dmbtr.
        MODIFY gt_standard FROM gs_standard TRANSPORTING month_dmbtr WHERE line_no = gs_standard-line_no.
      WHEN 'lastmonth_dmbtr'.
        gs_standard-lastmonth_dmbtr = ls_temp-month_dmbtr + gs_standard-lastmonth_dmbtr.
        MODIFY gt_standard FROM gs_standard TRANSPORTING lastmonth_dmbtr WHERE line_no = gs_standard-line_no.
      WHEN 'lastyear_dmbtr'.
        gs_standard-lastyear_dmbtr = ls_temp-month_dmbtr + gs_standard-lastyear_dmbtr.
