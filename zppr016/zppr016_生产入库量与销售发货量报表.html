FORM SELECT_dATA.

  SELECT mseg~werks mseg~matnr
         mseg~budat_mkpf
         mseg~aufnr
         mseg~shkzg
         mseg~menge
         mseg~bwart
         mseg~sobkz

         mara~bismt
         marc~fevor
         marc~dispo
    INTO CORRESPONDING FIELDS OF TABLE gt_mseg
    FROM mseg
    INNER JOIN mara ON mara~matnr = mseg~matnr
    INNER JOIN marc ON marc~matnr = mseg~matnr
                   AND marc~werks = mseg~werks
    WHERE mseg~werks = p_werks
      AND mseg~matnr IN s_matnr
      AND mseg~bwart IN s_bwart
      AND mseg~budat_mkpf IN s_DATUM
      AND mara~bismt IN s_bismt
      AND marc~fevor IN s_fevor
      AND marc~dispo IN s_dispo.


  DATA Lt_mseg TYPE TABLE OF ty_mseg.
  CLEAR:lt_mseg.
  lt_mseg[] = gt_mseg[].
  SORT lt_mseg BY matnr.
  DELETE ADJACENT DUPLICATES FROM lt_mseg COMPARING matnr.
  IF lt_mseg[] IS NOT INITIAL.
    SELECT *
      INTO CORRESPONDING FIELDS OF TABLE gt_makt
      FROM makt
      FOR ALL ENTRIES IN lt_mseg
      WHERE matnr = lt_mseg-matnr
        AND spras = sy-langu.
  ENDIF.

ENDFORM.

*& Form process_data
*&---------------------------------------------------------------------*
*& text
*&---------------------------------------------------------------------*
FORM process_data.

  SORT gt_makt BY matnr.

  LOOP AT gt_mseg INTO gs_mseg.
    CLEAR:gs_out.
*
    CASE 'X'.
      WHEN p_d.
        gs_out-date = gs_mseg-budat_mkpf.
      WHEN p_w.
        PERFORM get_date USING gs_mseg-budat_mkpf CHANGING gs_out-date.
      WHEN p_m.
        gs_out-date = gs_mseg-budat_mkpf+0(6).
    ENDCASE.

*
    gs_out-werks = gs_mseg-werks.
    gs_out-matnr = gs_mseg-matnr.
    gs_out-bismt = gs_mseg-bismt.
    gs_out-dispo = gs_mseg-dispo.
    gs_out-fevor = gs_mseg-fevor.

    CASE gs_mseg-bwart.
      WHEN '101' OR '102' OR '122' OR '123'.  "1. 生产入库数量

        IF gs_mseg-shkzg = 'H'.
          gs_out-grsum = gs_mseg-menge * -1.
        ELSE.
          gs_out-grsum = gs_mseg-menge.
        ENDIF.

      WHEN '601' OR '602' OR '631' OR '632' OR '653'. "2. 销售出货量

        IF gs_mseg-sobkz = ''.

          IF gs_mseg-shkzg = 'S'.
            gs_out-dlsum = gs_mseg-menge * -1.
          ELSE.
            gs_out-dlsum = gs_mseg-menge.
          ENDIF.

        ENDIF.

    ENDCASE.


    COLLECT gs_out INTO gt_out.
  ENDLOOP.

*
  LOOP AT gt_out ASSIGNING FIELD-SYMBOL(<fs_out>).
    CLEAR:gS_MAKT.
    READ TABLE gt_makt INTO gs_makt
         WITH KEY matnr = <fs_out>-matnr BINARY SEARCH.

    <fs_out>-maktx = gs_makt-maktx.

    IF <fs_out>-grsum IS INITIAL AND <fs_out>-dlsum IS INITIAL.
      <fs_out>-percent = 0.
    ELSEIF <fs_out>-grsum < 0 OR <fs_out>-dlsum < 0.
      <fs_out>-percent = ''.
    ELSEIF <fs_out>-grsum IS NOT INITIAL.
      <fs_out>-percent = <fs_out>-dlsum / <fs_out>-grsum * 100.
    ENDIF.

  ENDLOOP.

ENDFORM.

FORM modify_screen.

  LOOP AT SCREEN.
    CASE 'X'.
      WHEN p_D. "日
        IF screen-name CS 'S_WEEKD'
        OR screen-name CS 'S_MONTH'.
          screen-active = 0.
          MODIFY SCREEN.
        ENDIF.
      WHEN p_W. "周
        IF screen-name CS 'S_BUDAT'
        OR screen-name CS 'S_MONTH'.
          screen-active = 0.
          MODIFY SCREEN.
        ENDIF.
      WHEN p_M. "月
        IF screen-name CS 'S_BUDAT'
        OR screen-name CS 'S_WEEKD'.
