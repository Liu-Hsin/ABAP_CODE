FORM select_data.
*MRP控制者描述
  CLEAR:gt_t024d.
  SELECT *
    INTO TABLE gt_t024d
    FROM t024d
    WHERE werks = p_werks.

*生产管理员描述
  CLEAR:gt_t024f.
  SELECT *
    INTO TABLE gt_t024f
    FROM t024f
    WHERE werks = p_werks.


  SELECT afko~aufnr
         afko~rsnum
         afpo~matnr
         marc~dispo
         marc~fevor
         afpo~krsnr
         afpo~krsps
    INTO CORRESPONDING FIELDS OF TABLE gt_afpo
    FROM afpo
    INNER JOIN afko ON afko~aufnr = afpo~aufnr
    INNER JOIN aufk ON aufk~aufnr = afpo~aufnr
    INNER JOIN jest ON jest~objnr = aufk~objnr
    INNER JOIN marc ON marc~matnr = afpo~matnr
    WHERE afpo~pwerk = p_werks
      AND afpo~aufnr IN s_aufnr
      AND afpo~matnr IN s_matnr
      AND afpo~dauat IN s_dauat
      AND afko~gstrp IN s_gstrp
      AND afko~gltrp IN s_gltrp
      AND marc~dispo IN s_dispo
      AND marc~fevor IN s_fevor
      AND jest~stat = 'I0002'
      AND jest~inact = ''.


  IF gt_afpo[] IS NOT INITIAL.
    SELECT rsnum rspos
           rsart aufnr
           matnr
           meins
           bdter  "需求日期
           bdmng   "总需求量
           enmng   "已发货量
      INTO CORRESPONDING FIELDS OF TABLE gt_resb
      FROM resb
      FOR ALL ENTRIES IN gt_afpo
        WHERE rsnum = gt_afpo-rsnum
    AND
    (
    ( bdart = 'AR'
    AND xwaok = 'X'
    AND xloek = ''
    AND dumps = ''
    AND shkzg = 'H'
    AND kzear = '')
    OR
    ( bdart = 'BB'
    AND xloek = ''
    AND ebeln NE '')
    ).

*      WHERE rsnum = gt_afpo-krsnr
*        AND rspos = gt_afpo-krsps.
  ENDIF.


**********************************************************************
*       2022/3/22 XX-LIUX
*  委外订单预留逻辑
* 根据屏幕输入工厂，查询表RESB中，需求类型为BB（RESB-BDART = AR），
* 未删除（RESB-XLOEK =空），未最后发货（RESB-KZEAR = 空），借贷标识为H（RESB-SHKZG = H）的预留行，
* 并获取物料（RESB-MATNR）的MRP控制者（MARC-DISPO）为“M03”的行。下面逻辑与订单获取预留相同。
  SELECT rsnum,rspos,                                                 "
  rsart,aufnr,
  matnr,
  meins,
  bdter,  "需求日期
  bdmng,   "总需求量
  enmng   "已发货量
    INTO TABLE @DATA(lt_data)
    FROM resb
    WHERE werks = @p_werks
    AND bdart = 'BB'
    AND xloek = ''
    AND kzear = ''
    AND shkzg = 'H'.

  APPEND LINES OF lt_data TO gt_resb.
**********************************************************************


  DATA lt_resb TYPE TABLE OF ty_resb.
  CLEAR:lt_resb[].
  lt_resb[] = gt_resb[].
  SORT lt_resb BY matnr.
  DELETE ADJACENT DUPLICATES FROM lt_resb COMPARING matnr.
  IF lt_resb[] IS NOT INITIAL.
    SELECT marc~werks
           marc~matnr
           marc~dzeit
           marc~webaz
           marc~eisbe
           marc~dispo
           marc~fevor
           mara~bismt

      FROM marc
      INNER JOIN mara ON mara~matnr = marc~matnr
      INNER JOIN makt ON makt~matnr = mara~matnr
      FOR ALL ENTRIES IN lt_resb
      WHERE marc~matnr = lt_resb-matnr
        AND marc~werks = p_werks
        AND makt~spras = sy-langu
        AND marc~dispo = 'M03'.
  ENDIF.


ENDFORM.

FORM process_data.

  SORT gt_marc  BY matnr.
  SORT gt_t024d BY dispo.
  SORT gt_t024f BY fevor.

  LOOP AT gt_resb INTO gs_resb.
    CLEAR:gs_out.
    gs_out-werks = p_werks.
    gs_out-matnr = gs_resb-matnr.
    gs_out-bdter = gs_resb-bdter.
    gs_out-meins = gs_resb-meins.
    gs_out-opnquan = gs_resb-bdmng - gs_resb-enmng.
    COLLECT gs_out INTO gt_out.
  ENDLOOP.



  LOOP AT gt_out ASSIGNING <fs_out>.
    CLEAR:gs_marc.
    READ TABLE gt_marc INTO gs_marc
         WITH KEY matnr = <fs_out>-matnr BINARY SEARCH.
    IF sy-subrc <> 0.
      CLEAR:<fs_out>-matnr.
      CONTINUE.
    ENDIF.

    <fs_out>-bismt = gs_marc-bismt.
    <fs_out>-maktx = gs_marc-maktx.
    <fs_out>-dispo = gs_marc-dispo.
    <fs_out>-fevor = gs_marc-fevor.
    <fs_out>-dzeit = gs_marc-dzeit.
    <fs_out>-webaz = gs_marc-webaz.
    <fs_out>-eisbe = gs_marc-eisbe.

    CLEAR:gs_t024d.
    READ TABLE gt_t024d INTO gs_t024d
           WITH KEY dispo = <fs_out>-dispo BINARY SEARCH.
    <fs_out>-dsnam = gs_t024d-dsnam.

    CLEAR:gs_t024f.
    READ TABLE gt_t024f INTO gs_t024f
         WITH KEY fevor = <fs_out>-fevor BINARY SEARCH.
    <fs_out>-txt = gs_t024f-txt.

  ENDLOOP.


  DELETE gt_out WHERE matnr IS INITIAL.

ENDFORM.
