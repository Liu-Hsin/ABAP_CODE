FORM select_bom_single.

  SELECT * INTO CORRESPONDING FIELDS OF itab
    FROM mast WHERE matnr IN s_matnr
    AND werks IN s_werks AND stlan IN s_stlan AND stlal IN s_stlal.
*WHERE matnr IN s_matnr.

    CONCATENATE itab-mandt m itab-stlnr INTO key1.

    SELECT *
      FROM cdhdr
      APPENDING CORRESPONDING FIELDS OF TABLE itab_bom2
      WHERE objectclas = 'STUE'
        AND objectid = key1
        AND udate IN s_andat AND tcode <> 'CS01'.
    APPEND itab.
  ENDSELECT. "测试循环

  IF itab[] IS INITIAL.
    MESSAGE '没有数据.' TYPE 'E'.
  ENDIF.

FORM getdata .

*通过输入值，取出数据到itab_bom6中

*test add and modify
  LOOP AT itab_bom2.

    WHERE objectclas = 'STUE'
    AND changenr = itab_bom2-changenr
    AND objectid = itab_bom2-objectid.
      APPEND itab_bom6.
    ENDSELECT.
  ENDLOOP.
  DATA:it_stas LIKE TABLE OF stas WITH HEADER LINE.
*  it_stpo like table of stpo with header line .
  DATA:BEGIN OF it_stpo OCCURS 0,
      vgknt TYPE stpo-vgknt,mtart TYPE mara-mtart, posnr TYPE stpo-posnr, maktx TYPE makt-maktx, werks TYPE mast-werks,

       END OF it_stpo.
  LOOP AT itab_bom6.
    DATA: key12 TYPE stas-stasz.
    key11 = itab_bom6-objectid+4(8).
    IF itab_bom6-tabname = 'STPO'.
      key2 = itab_bom6-tabkey+12(8).
    ELSEIF itab_bom6-tabname = 'STAS'.
      key2 = itab_bom6-tabkey+14(8).
      k_stlal = itab_bom6-tabkey+12(2) .
*      CONCATENATE '*' itab_bom6-tabkey+12(2) INTO k_stlal.
      key12 = itab_bom6-tabkey+22(8).
    ENDIF.

    READ TABLE it_stas WITH KEY stlnr = key11  stlal = k_stlal stlkn = key2 BINARY SEARCH.
    IF sy-subrc = 0.
      MOVE-CORRESPONDING it_stas TO itab_bom6.
    ELSE.
      SELECT  *
*                 INTO CORRESPONDING FIELDS OF itab_bom6
                   INTO CORRESPONDING FIELDS OF it_stas
                   FROM stas
*            WHERE stlkn = key2 AND stlnr = key11 AND stasz = key12
*            AND stlal = k_stlal .
             WHERE
*             stlty IN ('D','E','K','M','S','T','P') AND
             stlty = 'M' and
             stlnr = key11 AND stlal = k_stlal AND stlkn = key2.
      ENDSELECT.
      IF sy-subrc = 0.
        MOVE-CORRESPONDING it_stas TO itab_bom6.
        APPEND it_stas.
        SORT it_stas BY stlnr stlal stlkn.
      ENDIF.
    ENDIF.

    READ TABLE it_stpo WITH KEY stlnr = key11 stlkn = key2 BINARY SEARCH.
    IF sy-subrc = 0.
      MOVE-CORRESPONDING it_stpo TO itab_bom6.
      IF itab_bom6-chngind = 'I' AND itab_bom6-vgknt = 0.
        MOVE itab_bom6-menge TO itab_bom6-value_new .
        CONDENSE itab_bom6-value_new .
        CONCATENATE '组件：' itab_bom6-idnrk  '数量：'
        itab_bom6-value_new INTO itab_bom6-value_new.
      ENDIF.
      MODIFY itab_bom6.
    ELSE.
      SELECT vgknt mtart posnr maktx werks stlan stpo~stlnr stlkn idnrk menge
      sortf potx1 potx2 datuv aennr
*               INTO CORRESPONDING FIELDS OF itab_bom6
               INTO CORRESPONDING FIELDS OF it_stpo
               FROM stpo
               INNER JOIN makt ON ( makt~matnr = stpo~idnrk and spras = sy-langu )
               INNER JOIN mara ON ( mara~matnr = stpo~idnrk )
               INNER JOIN mast ON ( mast~stlnr = stpo~stlnr )
               WHERE
*               stpo~stlty IN ('D','E','K','M','S','T','P') AND
               stlty = 'M' and
               stpo~stlnr = key11 AND stpo~stlkn = key2.
      ENDSELECT.
      IF sy-subrc = 0.
        MOVE-CORRESPONDING it_stpo TO itab_bom6.
        APPEND it_stpo.
        SORT it_stpo BY stlnr stlkn.
        IF itab_bom6-chngind = 'I' AND itab_bom6-vgknt = 0.
          MOVE itab_bom6-menge TO itab_bom6-value_new .
          CONDENSE itab_bom6-value_new .
          CONCATENATE '组件：' itab_bom6-idnrk  '数量：'
          itab_bom6-value_new INTO itab_bom6-value_new.
        ENDIF.
        MODIFY itab_bom6.
      ENDIF.
    ENDIF.
