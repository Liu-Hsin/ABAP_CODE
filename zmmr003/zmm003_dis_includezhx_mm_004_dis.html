*&---------------------------------------------------------------------*
*&      Form  FRM_GET_LTEXT
*&---------------------------------------------------------------------*
*       text
*----------------------------------------------------------------------*
*      -->P_LV_ID  text
*      -->P_LV_NAME  text
*      -->P_LV_OBJECT  text
*      <--P_LV_STR  text
*----------------------------------------------------------------------*
FORM frm_get_ltext USING p_id p_name p_object CHANGING p_str.
  DATA:lt_lines  TYPE tline OCCURS 0 WITH HEADER LINE,
       lv_id     TYPE thead-tdid,
       lv_name   TYPE thead-tdname,
       lv_object TYPE thead-tdobject.

  lv_id     = p_id .
  lv_name   = p_name.
  lv_object = p_object.

  CLEAR:p_str.

  CALL FUNCTION 'READ_TEXT'
    EXPORTING
      client                  = sy-mandt
      id                      = lv_id
      language                = sy-langu
      name                    = lv_name
      object                  = lv_object
    TABLES
      lines                   = lt_lines
    EXCEPTIONS
      id                      = 1
      language                = 2
      name                    = 3
      not_found               = 4
      object                  = 5
      reference_check         = 6
      wrong_access_to_archive = 7
      OTHERS                  = 8.
  IF sy-subrc <> 0.
  ENDIF.

  LOOP AT lt_lines.
    CONCATENATE p_str lt_lines-tdline INTO p_str.
  ENDLOOP.
ENDFORM.
