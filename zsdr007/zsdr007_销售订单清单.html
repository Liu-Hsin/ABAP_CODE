  TRY.
      CALL METHOD cl_salv_bs_runtime_info=>get_data_ref
        IMPORTING
          r_data = lr_data.

      ASSIGN lr_data->* TO <lt_data>.

  ENDTRY.

  ENDIF.

FORM PROCESS_dATA.

  LOOP AT gt_out ASSIGNING FIELD-SYMBOL(<fs_out>).

    SELECT SINGLE zshgc zshkw
      INTO (<fs_out>-zshgc,<fs_out>-zshkw)
      FROM vbap
      WHERE vbeln = <fs_out>-vbeln
        AND posnr = <fs_out>-posnr.

    SELECT SINGLE kdmat
      INTO <fs_out>-kdmat
      FROM knmt
      WHERE vkorg = <fs_out>-vkorg
*        AND vtweg = <fs_out>-vtweg
        AND kunnr = <fs_out>-kunnr
        AND matnr = <fs_out>-matnr.

  ENDLOOP.

ENDFORM.
