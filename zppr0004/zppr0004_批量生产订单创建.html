FORM upload_data .
  DATA:lt_excel TYPE TABLE OF  alsmex_tabline.
  DATA ls_excel LIKE LINE OF lt_excel.
  DATA:BEGIN OF gs_eine,
         matnr     TYPE marc-matnr,  " 物料
         bismt     TYPE mara-bismt,  " 图号
         maktx     TYPE makt-maktx,  " 物料描述
         fevor     TYPE marc-fevor,  " 生产管理员
         txt       TYPE t024f-txt,   " 生产管理员描述
         verid     TYPE mkal-verid,  " 生产版本
         gstrp     TYPE afko-gstrp,  " 生产开始日期
         gltrp     TYPE afko-gltrp,  " 生产结束日期
         gamng     TYPE afko-gamng , " 订单数量
         aprio     TYPE afko-aprio , " 优先级
         aptxt(10) TYPE c ,          " 优先级描述
       END OF gs_eine.
  DATA: gt_eine  LIKE TABLE OF gs_eine.
  DATA: lt_eine  LIKE TABLE OF gs_eine.




*  DATA: BEGIN OF i_data OCCURS 0.
*          INCLUDE STRUCTURE zfit006.
*  DATA END OF i_data.
*END OF i_data.
*  DATA: i_data LIKE TABLE OF gs_out.
  FIELD-SYMBOLS:<fs>.
**上传数据
  CALL FUNCTION 'ALSM_EXCEL_TO_INTERNAL_TABLE'
    EXPORTING
      filename                = p_file
      i_begin_col             = 1
      i_begin_row             = 2
      i_end_col               = 99
      i_end_row               = 9999
    TABLES
      intern                  = lt_excel
    EXCEPTIONS
      inconsistent_parameters = 1
      upload_ole              = 2
      OTHERS                  = 3.
  IF sy-subrc <> 0.
  ENDIF.
  LOOP AT lt_excel INTO ls_excel.
    ASSIGN COMPONENT ls_excel-col OF STRUCTURE gs_eine TO <fs>.
    CONDENSE ls_excel-value.                  " 去掉空格
    <fs> = ls_excel-value.
    AT END OF row.
      MOVE-CORRESPONDING gs_eine TO gs_out.
      CALL FUNCTION 'CONVERSION_EXIT_MATN1_INPUT'
        EXPORTING
          input  = gs_out-matnr
        IMPORTING
          output = gs_out-matnr.
      IF sy-subrc <> 0.
* Implement suitable error handling here
      ENDIF.
      APPEND gs_out TO  lt_out .
      CLEAR: gs_eine , gs_out.
    ENDAT.
  ENDLOOP.
  SELECT *  INTO TABLE @DATA(lt_marc)
     FROM marc
    FOR ALL ENTRIES IN @lt_out
      WHERE matnr = @lt_out-matnr
  AND werks = @s_pwerk .
  SELECT *  INTO TABLE @DATA(lt_t024f)
   FROM t024f
  FOR ALL ENTRIES IN @lt_marc
    WHERE fevor = @lt_marc-fevor
  AND werks = @s_pwerk .

  SELECT *  INTO TABLE @DATA(lt_mara)
   FROM mara
  FOR ALL ENTRIES IN @lt_marc
  WHERE matnr = @lt_marc-matnr.

  SELECT *  INTO TABLE @DATA(lt_makt)
   FROM makt
  FOR ALL ENTRIES IN @lt_marc
    WHERE matnr = @lt_marc-matnr
  AND spras = '1'.

  SELECT *  INTO TABLE @DATA(lt_mkal)
   FROM mkal
  FOR ALL ENTRIES IN @lt_out
    WHERE matnr = @lt_out-matnr
      AND werks = @s_pwerk
      AND verid = @lt_out-verid
      AND adatu <= @lt_out-gstrp
  AND bdatu >= @lt_out-gstrp .

  LOOP AT lt_out INTO gs_out.
    gs_out-pwerk = s_pwerk.
    gs_out-dauat = s_dauat.
    IF gs_out-gstrp IS INITIAL OR gs_out-gltrp IS INITIAL.
      gs_out-zbs = 'E'.
      gs_out-zmsgtxt = '日期为空或格式不正确，请调整！'.
    ELSE.
      CALL FUNCTION 'DATE_CHECK_PLAUSIBILITY'
        EXPORTING
          date = gs_out-gstrp.
      IF sy-subrc <> 0.
        gs_out-zbs = 'E'.
        gs_out-zmsgtxt = '日期为空或格式不正确，请调整！'.
      ENDIF.
      CALL FUNCTION 'DATE_CHECK_PLAUSIBILITY'
        EXPORTING
          date = gs_out-gltrp.
      IF sy-subrc <> 0.
        gs_out-zbs = 'E'.
        gs_out-zmsgtxt = '日期为空或格式不正确，请调整！'.
      ENDIF.
    ENDIF.
    IF gs_out-matnr IS INITIAL.
      gs_out-zbs = 'E'.
      gs_out-zmsgtxt = '物料编码为必输，请检查文档！'.
    ELSE.
      READ TABLE lt_marc INTO DATA(ls_marc) WITH KEY matnr = gs_out-matnr.
      IF sy-subrc <> 0.
        gs_out-zbs = 'F'.
        gs_out-zmsgtxt = '该物料在工厂' && gs_out-matnr && '下不存在，请检查！'.
      ELSE.
        gs_out-fevor = ls_marc-fevor.
        READ TABLE lt_t024f INTO DATA(ls_t024f) WITH KEY fevor = ls_marc-fevor.
        IF sy-subrc = 0.
          gs_out-txt = ls_t024f-txt.
        ENDIF.
      ENDIF.
      READ TABLE lt_mara INTO DATA(ls_mara) WITH KEY matnr = gs_out-matnr.
      IF sy-subrc = 0.
        gs_out-bismt = ls_mara-bismt.
      ENDIF.
      READ TABLE lt_makt INTO DATA(ls_makt) WITH KEY matnr = gs_out-matnr.
      IF sy-subrc = 0.
        gs_out-maktx = ls_makt-maktx.
      ENDIF.
      IF gs_out-verid IS INITIAL.
        gs_out-zbs = 'E'.
        gs_out-zmsgtxt = '生产版本为空，请维护！'.
      ELSE.
        IF lt_mkal IS INITIAL.
          gs_out-zbs = 'E'.
          gs_out-zmsgtxt = '该生产版本不存在或日期内无效，请检查！'.
        ENDIF.
      ENDIF.
    ENDIF.
    IF gs_out-gamng IS INITIAL.
      gs_out-zbs = 'E'.
      gs_out-zmsgtxt = '数量不可为空，请检查！'.
    ENDIF.
    IF  gs_out-aprio IS INITIAL.
      gs_out-zbs = 'E'.
      gs_out-zmsgtxt = '优先级维护错误，请检查！'.
    ELSE.
      CASE gs_out-aprio.
        WHEN 'A'.
          gs_out-aptxt = '特急'.
        WHEN 'B'.
          gs_out-aptxt = '紧急'.
        WHEN 'C'.
          gs_out-aptxt = '一般'.
        WHEN OTHERS.
          gs_out-zbs = 'E'.
          gs_out-zmsgtxt = '优先级维护错误，请检查！'.
      ENDCASE.
    ENDIF.
    APPEND gs_out TO gt_out.
    CLEAR:gs_out.
  ENDLOOP.


ENDFORM.
