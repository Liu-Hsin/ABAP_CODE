FORM modify_screen.

LOOP AT SCREEN.
    CASE screen-name.
      WHEN 'P_R3'.
        screen-input = 0.
      WHEN OTHERS.
    ENDCASE.
    MODIFY SCREEN.
  ENDLOOP.

  LOOP AT SCREEN.
    CASE 'X'.
      WHEN p_r1. "创建日志查询

      WHEN p_r2. "手动订单创建
        IF screen-name CS 'S_AUFNR'
          OR screen-name CS 'S_FEVOR'
          OR screen-name CS 'S_ERNAM'
          OR screen-name CS 'S_ERDAP'
          OR screen-name CS 'S_DISPO'.
          screen-active = 0.
          MODIFY SCREEN.
        ENDIF.
      WHEN p_r3. "自动订单创建（后台）
        IF screen-name CS 'S_AUFNR'
          OR screen-name CS 'S_FEVOR'
          OR screen-name CS 'S_ERNAM'
          OR screen-name CS 'S_ERDAP'
          OR screen-name CS 'S_DISPO'.
          screen-active = 0.

          MODIFY SCREEN.
        ENDIF.
    ENDCASE.
  ENDLOOP.

ENDFORM.

FORM select_data.

  SELECT vbap~werks  vbap~vbeln
         vbap~posnr  vbak~erdat
         vbak~auart  "vbak~edatu
         vbak~bstnk  vbak~kunnr
         vbap~kwmeng vbap~meins
         vbap~matnr
    INTO CORRESPONDING FIELDS OF TABLE gt_out
    FROM vbak
    INNER JOIN vbap ON vbap~vbeln = vbak~vbeln
    WHERE vbap~werks = p_werks
      AND vbak~auart IN s_auart
      AND vbak~kunnr IN s_kunnr
      AND vbak~erdat IN s_ERDAT
      AND vbap~vbeln IN s_vbeln
      AND vbap~posnr IN s_posnr
      AND vbap~lfsta = 'A'
      AND vbap~abgru = ''.

*排除在这个表： ZPPT004
  DATA lt_out LIKE TABLE OF ty_out.
  CLEAR:lt_out.
  lt_out[] = gt_out[].
  SORT lt_out BY vbeln.
  DELETE ADJACENT DUPLICATES FROM lt_out COMPARING vbeln.
  IF lt_out[] IS NOT INITIAL.
    SELECT *
      INTO CORRESPONDING FIELDS OF TABLE gt_zppt004
      FROM zppt004
      FOR ALL ENTRIES IN lt_out
      WHERE vbeln = lt_out-vbeln.
  ENDIF.

  LOOP AT gt_zppt004 INTO gs_zppt004.
    DELETE gt_out WHERE vbeln = gs_zppt004-vbeln
                    AND posnr = gs_zppt004-posnr.
  ENDLOOP.


*
  SELECT *
    INTO TABLE gt_t024f
    FROM t024f
    WHERE werks = p_werks.


*
  CLEAR:lt_out.
  lt_out[] = gt_out[].
  SORT lt_out BY matnr.
  DELETE ADJACENT DUPLICATES FROM lt_out COMPARING matnr.
  IF lt_out[] IS NOT INITIAL.

    CLEAR:gt_marc.
    SELECT matnr werks disgr
           dispo fevor dzeit
           webaz
      INTO CORRESPONDING FIELDS OF TABLE gt_marc
      FROM marc
      FOR ALL ENTRIES IN lt_out
      WHERE matnr = lt_out-matnr
        AND werks = p_werks
        AND dispo IN s_dispo
        AND fevor IN s_fevor.

    CLEAR:gt_mkal.
    SELECT *
      INTO TABLE gt_mkal
      FROM mkal
      FOR ALL ENTRIES IN lt_out
     WHERE matnr = lt_out-matnr
       AND werks = p_werks
       AND prfg_f = '1'.

    SELECT *
      INTO TABLE gt_makt
      FROM makt
      FOR ALL ENTRIES IN lt_out
      WHERE matnr = lt_out-matnr
        AND spras = sy-langu.

    SELECT matnr bismt
      INTO CORRESPONDING FIELDS OF TABLE gt_mara
      FROM mara
      FOR ALL ENTRIES IN lt_out
      WHERE matnr = lt_out-matnr
        AND bismt IN s_bismt.

  ENDIF.

ENDFORM.

FORM process_data.

  SORT gt_marc BY matnr.
  SORT gt_MKAL BY matnr.
  SORT gt_Mara BY matnr.
  SORT gt_MAKT BY matnr.
  SORT gt_t024f BY fevor.

  LOOP AT gt_out ASSIGNING <fs_out>.

* 最早交期（VBEP-EDATU）
    SELECT MIN( edatu )
      INTO <fs_out>-edatu
      FROM vbep
      WHERE vbeln = <fs_out>-vbeln
        AND posnr = <fs_out>-posnr.


    CLEAR:gs_mkal.
    READ TABLE gt_mkal INTO gs_mkal
          WITH KEY matnr = <fs_out>-matnr BINARY SEARCH.
    IF sy-subrc = 0.
      <fs_out>-verid = gs_mkal-verid.
    ELSE.

    ENDIF.

    CLEAR:gs_mara.
    READ TABLE gt_mara INTO gs_mara
         WITH KEY matnr = <fs_out>-matnr BINARY SEARCH.
    IF sy-subrc <> 0.
*不满足条件
      CLEAR:<fs_out>-matnr.
      CONTINUE.
    ENDIF.

    <fs_out>-bismt = gs_mara-bismt.


    CLEAR:gs_makt.
    READ TABLE gt_makt INTO gs_makt
         WITH KEY matnr = <fs_out>-matnr BINARY SEARCH.
    <fs_out>-maktx = gs_makt-maktx.

*
    CLEAR:gs_marc.
    READ TABLE gt_marc INTO gs_marc
       WITH KEY matnr = <fs_out>-matnr BINARY SEARCH.
    IF sy-subrc <> 0.
*不满足条件
      CLEAR:<fs_out>-matnr.
      CONTINUE.
    ENDIF.

    MOVE-CORRESPONDING gs_marc TO <fs_out>.

    CLEAR:gs_t024f.
    READ TABLE gt_t024f INTO gs_t024f
         WITH KEY fevor = <fs_out>-fevor BINARY SEARCH.
    <fs_out>-txt = gs_t024f-txt.


    CASE <fs_out>-disgr.
      WHEN '0020'.
        <fs_out>-aufart = 'Z007'.
      WHEN OTHERS.
        <fs_out>-aufart = 'Z001'.
    ENDCASE.

    <fs_out>-gltrp   = <fs_out>-edatu. "# 基本完成日期

    <fs_out>-gstrp   = <fs_out>-edatu. "# 基本开始日期

*生产提前期
    IF <fs_out>-webaz IS NOT INITIAL. "处理时间 WEBAZ

      PERFORM reduce_DATUM USING <fs_out>-gltrp "基本完成日期
                                 <fs_out>-webaz
                        CHANGING <fs_out>-gltrp.

      PERFORM reduce_DATUM USING <fs_out>-gstrp
                                 <fs_out>-webaz
                        CHANGING <fs_out>-gstrp.
    ENDIF.

    IF <fs_out>-dzeit IS NOT INITIAL. "自制时间

      PERFORM reduce_DATUM USING <fs_out>-gstrp
                                 <fs_out>-dzeit
                        CHANGING <fs_out>-gstrp.
    ENDIF.

    <fs_out>-ernam   = sy-uname. "# # 创建者
    <fs_out>-auerdat = sy-datum. "创建日期
    <fs_out>-aprio   = 'C'. "优先级


  ENDLOOP.

ENDFORM.

*& Form call_bapi_co01

*& text
*&---------------------------------------------------------------------*
FORM call_bapi_co01.

*检查是否重复创建
  DATA lv_ablad TYPE afpo-ablad.
  CLEAR:lv_ablad.

  lv_ablad = <fs_out>-vbeln && '/' && <fs_out>-posnr.

  SELECT SINGLE aufnr
    INTO <fs_out>-aufnr
    FROM afpo
    WHERE ablad = lv_ablad.
*&---------------------------------------------------------------------*
*&                  修改*&   20220308
  CLEAR: <fs_out>-status,
         <fs_out>-message.
*&---------------------------------------------------------------------*

  IF <fs_out>-aufnr IS NOT INITIAL.
    <fs_out>-message = '已存生产订单:' && <fs_out>-aufnr.
    <fs_out>-status  = icon_red_light.
  ENDIF.


*检查开始日期 和 结束日期是否小于 当前日期
  IF <fs_out>-gstrp < sy-datum.
    <fs_out>-message = '基本开始日期小于系统当前日期'.
    <fs_out>-status  = icon_red_light.
  ENDIF.

  IF <fs_out>-gltrp < sy-datum.
    <fs_out>-message = '基本结束日期小于系统当前日期'.
    <fs_out>-status  = icon_red_light.
  ENDIF.

  IF <fs_out>-verid IS INITIAL.
    <fs_out>-message = '生产版本为空'.
    <fs_out>-status  = icon_red_light.
  ENDIF.


  CHECK <fs_out>-status IS INITIAL.

*创建生产订单
  DATA ls_data   TYPE bapi_pp_order_create.
  DATA ls_return TYPE bapiret2.
  DATA lv_number TYPE bapi_order_key-order_number.
  DATA lv_type   TYPE bapi_order_copy-order_type.
  CLEAR:ls_data,ls_return,lv_number,lv_type.

  ls_data-material         = <fs_out>-matnr.
  ls_data-plant            = <fs_out>-werks.
  ls_data-order_type       = <fs_out>-aufart.
  ls_data-basic_start_date = <fs_out>-gstrp.
  ls_data-basic_end_date   = <fs_out>-gltrp.
  ls_data-quantity         = <fs_out>-kwmeng.
  ls_data-prod_version     = <fs_out>-verid.
  ls_data-order_priority   = <fs_out>-aprio.
  ls_data-unloading_point  = <fs_out>-vbeln && '/' && <fs_out>-posnr.

  IF <fs_out>-disgr = '0020'.
    ls_data-sales_order      = <fs_out>-vbeln.
    ls_data-sales_order_item = <fs_out>-posnr.
  ENDIF.

  CALL FUNCTION 'BAPI_PRODORD_CREATE'
    EXPORTING
      orderdata    = ls_data
    IMPORTING
      return       = ls_return
      order_number = lv_number
      order_type   = lv_type.

  IF ls_return-type = 'E'.

    CALL FUNCTION 'BAPI_TRANSACTION_ROLLBACK'.
    ROLLBACK WORK.
    <fs_out>-message = '创建失败:' && ls_return-message.
    <fs_out>-status  = icon_red_light.

  ELSE.

    <fs_out>-aufnr   = lv_number.
    <fs_out>-status  = icon_green_light.
    <fs_out>-message = '生产订单创建成功'.

    PERFORM save_zppt004.

    CALL FUNCTION 'BAPI_TRANSACTION_COMMIT'
      EXPORTING
        wait = 'X'.

  ENDIF.

ENDFORM.
