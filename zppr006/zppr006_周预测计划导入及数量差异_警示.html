FORM read_excel.

  DATA lv_filename TYPE rlgrap-filename.
  DATA :lv_index TYPE i.
  FIELD-SYMBOLS : <fs> TYPE any.
  CLEAR lv_filename.
  lv_filename = p_file.

  CALL FUNCTION 'ALSM_EXCEL_TO_INTERNAL_TABLE'
    EXPORTING
      filename                = lv_filename
      i_begin_col             = 1
      i_begin_row             = 1
      i_end_col               = 256
      i_end_row               = 65535
    TABLES
      intern                  = gt_excel
    EXCEPTIONS
      inconsistent_parameters = 1
      upload_ole              = 2
      OTHERS                  = 3.

  IF sy-subrc <> 0.
    MESSAGE ID sy-msgid TYPE sy-msgty NUMBER sy-msgno
            WITH sy-msgv1 sy-msgv2 sy-msgv3 sy-msgv4.
  ENDIF.

  SORT gt_excel BY row col.

  CLEAR:gt_data,gs_data.
  CLEAR:gs_datum,gt_datum.
*先取到第一行的日期


  LOOP AT gt_excel INTO gs_excel.

    IF gs_excel-row = 1. "抬头日期

      IF gs_excel-col = 1.
*物料号
      ELSE.
        CLEAR:gs_datum.
        gs_datum-col   = gs_excel-col.
        gs_datum-datum = gs_excel-value.
        APPEND gs_datum TO gt_datum.
      ENDIF.

    ELSE.

      IF gs_excel-col = 1.
        CLEAR:gs_data.
        gs_data-matnr = gs_excel-value.
*&---------------------------------------------------------------------*
*&                  程序修改 20220314 *&

        CALL FUNCTION 'CONVERSION_EXIT_MATN1_INPUT'
          EXPORTING
            input  = gs_data-matnr
          IMPORTING
            output = gs_data-matnr.
*&---------------------------------------------------------------------*
      ELSE.
        CLEAR:gs_datum.
        READ TABLE gt_datum INTO gs_datum WITH KEY col = gs_excel-col.
        gs_data-datum = gs_datum-datum.
        gs_data-menge = gs_excel-value.
        APPEND gs_data TO gt_data.
      ENDIF.

    ENDIF.

*    MOVE gs_excel-col TO lv_index.
*    ASSIGN COMPONENT lv_index OF STRUCTURE gs_out TO <fs>.
*    MOVE gs_excel-value TO <fs>.
*
*    AT END OF row.
*      APPEND gs_out TO gt_out.
*      CLEAR gs_out.
*    ENDAT.

  ENDLOOP.

ENDFORM.

*&---------------------------------------------------------------------*
FORM select_data.
  CLEAR:gt_zppt003.

  SELECT *
    INTO CORRESPONDING FIELDS OF TABLE gt_zppt003
    FROM zppt003
    WHERE werks = p_werks
      AND zversion IN s_versi
      AND zmatnr1 IN s_matnr1
      AND zmatnr2 IN s_matnr2
      AND zindctr IN s_zindct
      AND zplndat IN s_zplnda.

ENDFORM.

*&---------------------------------------------------------------------*
*& text
*&---------------------------------------------------------------------*
FORM process_data.

  CLEAR:gt_out.

  LOOP AT gt_zppt003 INTO gs_zppt003.
    CLEAR:gs_out.
    MOVE-CORRESPONDING gs_zppt003 TO gs_out.

*&---------------------------------------------------------------------*
*&                 程序修改 20220314 查询界面增加行颜色*&
    gs_out-zindctr_t = gs_out-zindctr. "差异标识
*I  增加
    IF gs_out-zindctr = 'I'.
      gs_out-zindctr_t = '增加'.
      gs_out-clr = 'C710'.
    ENDIF.
*R  减少
    IF gs_out-zindctr  = 'R'.
      gs_out-zindctr_t = '减少'.
      gs_out-clr = 'C610'.
    ENDIF.

*E  相等
    IF  gs_out-zindctr = 'E'.
      gs_out-zindctr_t = '相等'.

    ENDIF.

*A  新增
    IF gs_out-zindctr = 'A'.
      gs_out-clr = 'C510'.
      gs_out-zindctr_t = '新增'.
    ENDIF.
*D  删除
    IF gs_out-zindctr = 'D'.
      gs_out-zindctr_t = '删除'.
    ENDIF.
*&---------------------------------------------------------------------*

    APPEND gs_out TO gt_out.
  ENDLOOP.

ENDFORM.

FORM conversion_data.

*取最大版本号
  CLEAR:gv_zversion.
  SELECT SINGLE MAX( zversion )
    INTO gv_zversion
    FROM zppt003
    WHERE zdate = sy-datum
      AND werks = p_werks.

  IF gv_zversion <> '00000000000' AND sy-subrc = 0.
*    gv_zversion = gv_zversion+0(8) && ( GV_zversion+8(3) + 1 ).
    gv_zversion = gv_zversion + 1.
  ELSE.
    gv_zversion = sy-datum && '001'.
  ENDIF.



*GT_DATA
  DATA lt_data TYPE TABLE OF ty_data.
  CLEAR:lt_data.
  lt_data[] = gt_data[].
  SORT lt_data BY matnr.
  DELETE ADJACENT DUPLICATES FROM lt_data COMPARING matnr.
  IF lt_data[] IS NOT INITIAL.
    SELECT matnr werks dispo beskz
      INTO CORRESPONDING FIELDS OF TABLE gt_marc
      FROM marc
      FOR ALL ENTRIES IN lt_data
      WHERE matnr = lt_data-matnr
        AND werks = p_werks.

    SELECT mara~matnr
           mara~bismt
           mara~meins
           makt~maktx
      INTO CORRESPONDING FIELDS OF TABLE gt_mara
      FROM mara
      INNER JOIN makt ON makt~matnr = mara~matnr
      FOR ALL ENTRIES IN lt_data
      WHERE mara~matnr = lt_data-matnr
        AND makt~spras = sy-langu.
  ENDIF.

  SORT gt_marc BY matnr.
  SORT gt_mara BY matnr.

  CLEAR:gt_out.
  LOOP AT gt_data INTO gs_data.
    CLEAR:gs_out.
    CLEAR:gs_marc.
    READ TABLE gt_marc INTO gs_marc WITH KEY matnr = gs_data-matnr BINARY SEARCH.

    CLEAR:gs_mara.
    READ TABLE gt_mara INTO gs_mara WITH KEY matnr = gs_data-matnr BINARY SEARCH.

    CASE gs_marc-dispo.
      WHEN 'M00'.
*取BOM
        PERFORM call_bom USING gs_data-matnr gs_data-menge.

        LOOP AT gt_bom INTO gs_bom WHERE matnr = gs_data-matnr.
          gs_out-werks   = p_werks.
          gs_out-zmatnr1 = gs_bom-idnrk.
          gs_out-zmaktx1 = gs_bom-maktx.
          gs_out-zbismt1 = gs_bom-bismt.
          gs_out-beskz   = gs_bom-beskz.
          gs_out-zmatnr2 = gs_bom-matnr.
          gs_out-zbismt2 = gs_mara-bismt.
          gs_out-zmaktx2 = gs_mara-maktx.
          gs_out-zquanty = gs_bom-mnglg.
          gs_out-meins   = gs_bom-meins.

          gs_out-zplndat = gs_data-datum.
          gs_out-zdate   = sy-datum.
          gs_out-ztime   = sy-uzeit.
          gs_out-zernam  = sy-uname.
          gs_out-zversion = gv_zversion.
          APPEND gs_out TO gt_out.
        ENDLOOP.

      WHEN OTHERS.

        gs_out-zmatnr1 = gs_data-matnr.
        gs_out-zplndat = gs_data-datum.
        gs_out-werks   = p_werks.
        gs_out-zmaktx1 = gs_mara-maktx.
        gs_out-zbismt1 = gs_mara-bismt.
        gs_out-zquanty = gs_data-menge.
        gs_out-meins   = gs_mara-meins.
        gs_out-zdate   = sy-datum.
        gs_out-ztime   = sy-uzeit.
        gs_out-zernam  = sy-uname.
        gs_out-beskz   = gs_marc-beskz.
        gs_out-zversion = gv_zversion.
        APPEND gs_out TO gt_out.

    ENDCASE.

  ENDLOOP.


*对比上一版本
  DATA ls_zppt003 TYPE zppt003.
  DATA lt_zppt003 TYPE TABLE OF zppt003.
  CLEAR:lt_zppt003,ls_zppt003.

*ZINDCTR  差异标识
*ZPRCTGE 增加/减少百分比
*ZDIFQUAN 差异数量

*取最大版本号
  CLEAR:gv_zversion.
  SELECT SINGLE MAX( zversion )
    INTO gv_zversion
    FROM zppt003
    WHERE werks = p_werks.

  SELECT *
    INTO CORRESPONDING FIELDS OF TABLE lt_zppt003
    FROM zppt003
    WHERE werks = p_werks
      AND zversion = gv_zversion
      AND zindctr <> 'D'.



  SORT lt_zppt003 BY zmatnr1 zplndat zmatnr2.

*D  删除
  LOOP AT gt_out ASSIGNING <fs_out>.
    CLEAR:ls_zppt003.
    READ TABLE lt_zppt003 INTO ls_zppt003
          WITH KEY  zmatnr1 = <fs_out>-zmatnr1
                    zplndat = <fs_out>-zplndat
                    zmatnr2 = <fs_out>-zmatnr2 BINARY SEARCH.

    IF sy-subrc = 0.
*I  增加
      IF ls_zppt003-zquanty < <fs_out>-zquanty.
        <fs_out>-zindctr   = 'I'.
        <fs_out>-zindctr_t = '增加'.
        <fs_out>-zdifquan = <fs_out>-zquanty - ls_zppt003-zquanty.
        <fs_out>-zprctge  = <fs_out>-zdifquan * 100 / ls_zppt003-zquanty.
        <fs_out>-clr = 'C710'.
*R  减少
      ELSEIF ls_zppt003-zquanty > <fs_out>-zquanty.
        <fs_out>-zindctr  = 'R'.
        <fs_out>-zindctr_t = '减少'.
        <fs_out>-zdifquan = ls_zppt003-zquanty - <fs_out>-zquanty.
        <fs_out>-zprctge  = <fs_out>-zdifquan * 100 / ls_zppt003-zquanty.
        <fs_out>-clr = 'C610'.
*E  相等
      ELSEIF ls_zppt003-zquanty = <fs_out>-zquanty.
        <fs_out>-zindctr = 'E'.
        <fs_out>-zindctr_t = '相等'.
      ENDIF.

    ELSE.
*A  新增
      <fs_out>-clr = 'C510'.
      <fs_out>-zindctr_t = '新增'.
      <fs_out>-zindctr = 'A'.
      <fs_out>-zprctge = '100'.
    ENDIF.

  ENDLOOP.


*在 ZPPT003 里面有  GT_OUT 里面没有的显示 删除
  LOOP AT lt_zppt003 INTO ls_zppt003.

    CLEAR:gs_out.
    READ TABLE gt_out INTO gs_out
          WITH KEY  zmatnr1 = ls_zppt003-zmatnr1
                    zplndat = ls_zppt003-zplndat
                    zmatnr2 = ls_zppt003-zmatnr2.
    IF sy-subrc <> 0.
      CLEAR:gs_out.
      MOVE-CORRESPONDING ls_zppt003 TO gs_out.
      gs_out-clr = 'C410'.
      gs_out-zindctr_t = '删除'.
      gs_out-zindctr = 'D'.
      APPEND gs_out TO gt_out.
    ENDIF.

  ENDLOOP.



  SORT gt_out BY zmatnr1 zplndat.

ENDFORM.

FORM call_bom USING uv_matnr uv_menge.
  DATA lt_stb TYPE TABLE OF stpox.
  DATA ls_stb TYPE          stpox.
  DATA lt_mat TYPE TABLE OF cscmat.
  DATA ls_mat TYPE          cscmat.

  CLEAR:gt_bom.
*  CLEAR:gs_bom.
*  READ TABLE gt_bom INTO gs_bom WITH KEY matnr = uv_matnr.
*
*  CHECK sy-subrc <> 0.

  CALL FUNCTION 'CS_BOM_EXPL_MAT_V2'
    EXPORTING
      capid                 = 'PP01'          " BOM应用
      datuv                 = sy-datum          " 有效起始日
*     ehndl                 = '1'
      emeng                 = uv_menge          " 需求数量
      mehrs                 = ''               " 多层展开
      mmory                 = '1'               " 是否使用缓存
      mtnrv                 = uv_matnr         " 展开物料号
*     stlan                 = '1'               " BOM用途
      werks                 = p_werks          " 物料所在工厂
    TABLES
      stb                   = lt_stb
      matcat                = lt_mat
    EXCEPTIONS
      alt_not_found         = 1
      call_invalid          = 2
      material_not_found    = 3
      missing_authorization = 4
      no_bom_found          = 5
      no_plant_data         = 6
      no_suitable_bom_found = 7
      conversion_error      = 8
      OTHERS                = 9.


  LOOP AT lt_stb INTO ls_stb.
    CLEAR:gs_bom.
    gs_bom-matnr = uv_matnr.
    gs_bom-idnrk = ls_stb-idnrk.

    SELECT SINGLE meins bismt
      INTO ( gs_bom-meins,gs_bom-bismt )
      FROM mara
      WHERE matnr = ls_stb-idnrk.

    SELECT SINGLE maktx
      INTO gs_bom-maktx
      FROM makt
      WHERE matnr = ls_stb-idnrk
        AND spras = sy-langu.

    SELECT SINGLE  beskz
      INTO gs_bom-beskz
      FROM marc
      WHERE matnr = ls_stb-idnrk
        AND werks = p_werks.

    IF gs_bom-meins = ls_stb-mmein.
      gs_bom-mnglg = ls_stb-mnglg.
    ELSE.
      CALL FUNCTION 'MD_CONVERT_MATERIAL_UNIT'
        EXPORTING
          i_matnr              = ls_stb-idnrk
          i_in_me              = ls_stb-mmein
          i_out_me             = gs_bom-meins
          i_menge              = ls_stb-mnglg
        IMPORTING
          e_menge              = gs_bom-mnglg
        EXCEPTIONS
          error_in_application = 1
          error                = 2
          OTHERS               = 3.

    ENDIF.

    COLLECT gs_bom INTO gt_bom.

  ENDLOOP.

ENDFORM.

FORM save_data.
  DATA lt_zppt003 TYPE TABLE OF zppt003.
  DATA ls_zppt003 TYPE          zppt003.
  CLEAR:lt_zppt003,ls_zppt003.

  LOOP AT gt_out INTO gs_out.
    CLEAR:ls_zppt003.
    MOVE-CORRESPONDING gs_out TO ls_zppt003.
    APPEND ls_zppt003 TO lt_zppt003.
  ENDLOOP.

  IF lt_zppt003[] IS NOT INITIAL.
    MODIFY zppt003 FROM TABLE lt_zppt003.
    COMMIT WORK.
    MESSAGE '数据保存成功！' TYPE 'S'.
  ENDIF.

ENDFORM.
