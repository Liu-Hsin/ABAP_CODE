FORM reple_data .
  IF gt_p005[] IS NOT INITIAL.
    SELECT matnr, bismt
      INTO TABLE @DATA(lt_mara)
      FROM mara
      FOR ALL ENTRIES IN @gt_p005
      WHERE matnr = @gt_p005-matnr.

    SELECT auart, txt
      INTO TABLE @DATA(lt_t003p)
      FROM t003p
      FOR ALL ENTRIES IN @gt_p005
      WHERE auart = @gt_p005-dauat
        AND spras = 1.

    SELECT aufnr, lgort
      INTO TABLE @DATA(lt_afpo)
      FROM afpo
      FOR ALL ENTRIES IN @gt_p005
      WHERE aufnr = @gt_p005-aufnr.


  ENDIF.

  IF gt_p006[] IS NOT INITIAL.
    SELECT matnr, bismt
      INTO TABLE @DATA(lt_mara1)
      FROM mara
      FOR ALL ENTRIES IN @gt_p006
      WHERE matnr = @gt_p006-matnr.

  ENDIF.

  LOOP AT gt_p005 INTO gs_p005.
    READ TABLE lt_mara INTO DATA(ls_mara) WITH KEY matnr = gs_p005-matnr.
    IF sy-subrc = 0.
      gs_p005-bismt = ls_mara-bismt.

    ENDIF.

    READ TABLE lt_t003p INTO DATA(ls_t003p) WITH KEY auart = gs_p005-dauat.
    IF sy-subrc = 0.
      gs_p005-txt = ls_t003p-txt.

    ENDIF.

    READ TABLE lt_afpo INTO DATA(ls_afpo) WITH KEY aufnr = gs_p005-aufnr.
    IF sy-subrc = 0.
      gs_p005-lgort = ls_afpo-lgort.
    ENDIF.
    IF gs_p005-saprel = 'F'.
      gs_p005-zlight = icon_led_red.

      gs_p005-zlight = icon_led_green.
    ENDIF.
