FORM sub_upload_data .
  IF p_file IS INITIAL.
    MESSAGE e000 WITH '请输入上传文件路径!'.
  ENDIF.
  v_len = strlen( p_file ).
  v_len = v_len - 3.
  v_suffix = p_file+v_len(3).
  TRANSLATE v_suffix TO UPPER CASE.
  IF v_suffix <> 'LSX' AND v_suffix <> 'lsx'.
    MESSAGE e000 WITH '导入的文件名必须类型为"XLSX"的Excel文件!'.
  ENDIF.

  REFRESH it_input.
  CALL FUNCTION 'TEXT_CONVERT_XLS_TO_SAP'
    EXPORTING
      i_line_header        = 'X'
      i_tab_raw_data       = it_raw
      i_filename           = p_file
    TABLES
      i_tab_converted_data = it_input
    EXCEPTIONS
      conversion_failed    = 1
      OTHERS               = 2.
ENDFORM.


FORM sub_check_data_input .
  REFRESH it_show.
  LOOP AT it_input.
    CLEAR it_show.
    MOVE-CORRESPONDING it_input TO it_show.

    IF it_show-vkorg IS INITIAL.
      CONCATENATE it_show-ermsg '销售组织为空|' INTO it_show-ermsg.
    ELSE.
      SELECT SINGLE * INTO wa_tvko FROM tvko
        WHERE vkorg = it_show-vkorg.
      IF sy-subrc <> 0.
        CONCATENATE it_show-ermsg '销售组织无效|' INTO it_show-ermsg.
      ENDIF.
    ENDIF.

    IF it_show-vtweg IS INITIAL.
      CONCATENATE it_show-ermsg '分销渠道为空|' INTO it_show-ermsg.
    ELSE.
      SELECT SINGLE * INTO wa_tvtw FROM tvtw
        WHERE vtweg = it_show-vtweg.
      IF sy-subrc <> 0.
        CONCATENATE it_show-ermsg '分销渠道无效|' INTO it_show-ermsg.
      ENDIF.
    ENDIF.

    IF NOT it_show-kunnr IS INITIAL.
      CALL FUNCTION 'CONVERSION_EXIT_ALPHA_INPUT'
        EXPORTING
          input  = it_show-kunnr
        IMPORTING
          output = it_show-kunnr.
      SELECT SINGLE * INTO wa_kna1 FROM kna1
        WHERE kunnr = it_show-kunnr.
      IF sy-subrc <> 0.
        CONCATENATE it_show-ermsg '客户编码无效|' INTO it_show-ermsg.
      ELSE.
        SELECT SINGLE * INTO wa_knvv FROM knvv
          WHERE kunnr = it_show-kunnr
            AND vkorg = it_show-vkorg
            AND vtweg = '00'."it_show-vtweg.
        IF sy-subrc <> 0.
          CONCATENATE it_show-ermsg '客户未维护销售试图|' INTO it_show-ermsg.
        ENDIF.
      ENDIF.
    ENDIF.

    IF it_show-matnr IS INITIAL.
      CONCATENATE it_show-ermsg '物料编码为空|' INTO it_show-ermsg.
    ELSE.
      CALL FUNCTION 'CONVERSION_EXIT_MATN1_INPUT'
        EXPORTING
          input        = it_show-matnr
        IMPORTING
          output       = it_show-matnr
        EXCEPTIONS
          length_error = 1
          OTHERS       = 2.
      IF sy-subrc <> 0.
      ENDIF.
      SELECT SINGLE * INTO wa_mara FROM mara
        WHERE matnr = it_show-matnr.
      IF sy-subrc <> 0.
        CONCATENATE it_show-ermsg '物料编码无效|' INTO it_show-ermsg.
      ELSE.
        SELECT SINGLE * INTO wa_mvke FROM mvke
          WHERE matnr = it_show-matnr
            AND vkorg = it_show-vkorg
            AND vtweg = '00'."it_show-vtweg.
        IF sy-subrc <> 0.
          CONCATENATE it_show-ermsg '物料未维护销售试图|' INTO it_show-ermsg.
        ENDIF.
      ENDIF.
    ENDIF.

    IF it_show-kbetr_c IS INITIAL.
      CONCATENATE it_show-ermsg '含税价格为空|' INTO it_show-ermsg.
    ELSE.
      TRY.
          it_show-kbetr = it_show-kbetr_c.
        CATCH cx_sy_conversion_no_number.
          CONCATENATE it_show-ermsg '含税价格无效|' INTO it_show-ermsg.
      ENDTRY.
    ENDIF.

    IF it_show-konwa <> 'CNY' AND it_show-konwa <> 'USD'.
      CONCATENATE it_show-ermsg '币制无效|' INTO it_show-ermsg.
    ENDIF.

    IF it_show-kpein_c IS INITIAL.
      it_show-kpein = 1.
    ELSE.
      TRY.
          it_show-kpein = it_show-kpein_c.
        CATCH cx_sy_conversion_no_number.
          CONCATENATE it_show-ermsg '价格单位无效|' INTO it_show-ermsg.
      ENDTRY.
    ENDIF.

    IF it_show-kmein IS INITIAL.
      SELECT SINGLE meins INTO it_show-kmein FROM mara
        WHERE matnr = it_show-matnr.
    ELSE.
      SELECT SINGLE * INTO wa_t006 FROM t006
        WHERE msehi = it_show-kmein.
      IF sy-subrc <> 0.
        CONCATENATE it_show-ermsg '计量单位无效|' INTO it_show-ermsg.
      ENDIF.
    ENDIF.

    IF it_show-datab = '' OR it_show-datab IS INITIAL.
      it_show-datab = sy-datum.
    ELSE.
      CALL FUNCTION 'RP_CHECK_DATE'
        EXPORTING
          date         = it_show-datab
        EXCEPTIONS
          date_invalid = 1
          OTHERS       = 2.
      IF sy-subrc <> 0.
        CONCATENATE it_show-ermsg '有效期自无效|' INTO it_show-ermsg.
      ENDIF.
    ENDIF.

    IF it_show-datbi = '' OR it_show-datbi IS INITIAL.
      it_show-datbi = '99991231'.
    ELSE.
      CALL FUNCTION 'RP_CHECK_DATE'
        EXPORTING
          date         = it_show-datbi
        EXCEPTIONS
          date_invalid = 1
          OTHERS       = 2.
      IF sy-subrc <> 0.
        CONCATENATE it_show-ermsg '有效期至无效|' INTO it_show-ermsg.
      ENDIF.
    ENDIF.

    IF it_show-datbi < it_show-datab.
      CONCATENATE it_show-ermsg '有效期至小于有效期自|' INTO it_show-ermsg.
    ENDIF.

    IF it_show-ermsg IS INITIAL.
      it_show-icon = icon_yellow_light.
    ELSE.
      it_show-icon = icon_red_light.
      it_show-error = 'X'.
    ENDIF.

    APPEND it_show.
  ENDLOOP.
ENDFORM.
