FORM process_data_r1.
  TYPES:BEGIN OF Ly_head,
          box       TYPE c,

          zlabst    TYPE mard-labst,
          mtart     TYPE mara-mtart,
        END OF Ly_head.
  DATA Lt_head TYPE TABLE OF Ly_head.
  DATA Ls_head TYPE          Ly_head.



  CLEAR:gt_head[],gt_item[],gt_righ[].
  CLEAR:lt_head[].

  LOOP AT GT_zmmllpsjhd INTO GS_zmmllpsjhd.
*ITEM
    CLEAR:gs_item.
    MOVE-CORRESPONDING GS_zmmllpsjhd TO gs_item.
    APPEND gs_item TO gt_item.

*HEAD
    CLEAR:Ls_head.
    MOVE-CORRESPONDING GS_zmmllpsjhd TO Ls_HEAD.
    ls_head-zrxpsslhz = GS_zmmllpsjhd-zbcllsl.
    ls_head-zbcfhsl   = GS_zmmllpsjhd-zbcllsl.

    SELECT SINGLE mtart
      INTO ls_head-mtart
      FROM mara
      WHERE matnr = ls_head-matnr.

    COLLECT Ls_head INTO Lt_head.

  ENDLOOP.


*
  DATA:ls_fieldstyle TYPE lvc_s_styl.
  DATA:lt_fieldstyle TYPE  TABLE OF lvc_s_styl.

  LOOP AT Lt_head INTO ls_head.

*批次
    SELECT SINGLE xchar
      INTO Ls_head-xchar
      FROM marc
      WHERE matnr = Ls_head-matnr
        AND werks = Ls_head-werks.


*非限制库存
    CASE Ls_head-xchar.
      WHEN 'X'.     "启用批次
        SELECT SUM( clabs )
          INTO Ls_head-zlabst
          FROM mchb
          WHERE matnr = Ls_head-matnr
            AND werks = Ls_head-werks.
      WHEN OTHERS. "未启用批次
        SELECT SUM( clabs )
          INTO Ls_head-zlabst
          FROM mchb
          WHERE matnr = Ls_head-matnr
            AND werks = Ls_head-werks.
    ENDCASE.

    CLEAR:lt_fieldstyle[].
*控制单元格
    IF Ls_head-xchar = 'X'.
      CLEAR:ls_fieldstyle.
      ls_fieldstyle-fieldname = 'ZWIDTH'.
      ls_fieldstyle-style     = cl_gui_alv_grid=>mc_style_enabled.
      APPEND ls_fieldstyle TO lt_fieldstyle.

      CLEAR:ls_fieldstyle.
      ls_fieldstyle-fieldname = 'ZLENGTH'.
      ls_fieldstyle-style     = cl_gui_alv_grid=>mc_style_enabled.
      APPEND ls_fieldstyle TO lt_fieldstyle.
    ELSE.
      CLEAR:ls_fieldstyle.
      ls_fieldstyle-fieldname = 'ZWIDTH'.
      ls_fieldstyle-style     = cl_gui_alv_grid=>mc_style_disabled.
      APPEND ls_fieldstyle TO lt_fieldstyle.

      CLEAR:ls_fieldstyle.
      ls_fieldstyle-fieldname = 'ZLENGTH'.
      ls_fieldstyle-style     = cl_gui_alv_grid=>mc_style_disabled.
      APPEND ls_fieldstyle TO lt_fieldstyle.

    ENDIF.

    CLEAR:gs_head.
    CLEAR:Gs_head-field[].

    MOVE-CORRESPONDING ls_head TO gs_head.

    Gs_head-field[] = lt_fieldstyle[].
    APPEND GS_head TO GT_head.

  ENDLOOP.

ENDFORM.

FORM handle_data_changed_HEAD USING us_data_changed
                    TYPE REF TO cl_alv_changed_data_protocol.

*取批次特性
*  xchar = 'X'.
*宽  zwidth
*长  zlength


  DATA:x_change TYPE lvc_s_modi.

  LOOP AT us_data_changed->mt_good_cells INTO x_change.

    CASE x_change-fieldname.
      WHEN 'ZBCFHSL'. "本次发货数量
        PERFORM CHECK_zbcfhsl USING x_change-value x_change-row_id.
      WHEN 'LGORT'.
*检查库存地是否正确
        PERFORM check_lgort USING x_change-value x_change-row_id.

      WHEN 'ZWIDTH'. "宽 更新库存
        PERFORM check_Zwidth USING x_change-value x_change-row_id.

      WHEN 'ZLENGTH'. "长
        PERFORM check_zlength USING x_change-value x_change-row_id.
      WHEN OTHERS.
    ENDCASE.


*    CALL METHOD gv_grid_HEAD->refresh_table_display
*      EXCEPTIONS
*        finished = 1
*        OTHERS   = 2.

  ENDLOOP.


  LOOP AT gt_head ASSIGNING <fs_head>.

*更新库存
    IF <fs_head>-zwidth IS INITIAL
    AND <fs_head>-zlength IS INITIAL
    AND <fs_head>-lgort IS INITIAL.
      CONTINUE.
    ENDIF.

    CLEAR:<fs_head>-zlabst.
    PERFORM GET_zlabst USING <fs_head>-lgort
                             <fs_head>-matnr
                             <fs_head>-zwidth
                             <fs_head>-zlength
                             <fs_head>-xchar
                    CHANGING <fs_head>-zlabst.

    IF <fs_head>-zlabst < <fs_head>-zbcfhsl.
      <fs_head>-icon    = icon_red_light.
      <fs_head>-message = '库存小于本次发货数量'.
    ENDIF.

  ENDLOOP.

  CALL METHOD gv_grid_HEAD->refresh_table_display
    EXCEPTIONS
      finished = 1
      OTHERS   = 2.

ENDFORM.

    lv_mark = 'X'.
  ENDIF.

*检查库存是否满足


  CHECK lv_mark IS  INITIAL.
  TYPES:BEGIN OF ty_mchb,
          matnr TYPE mChb-matnr,
          werks TYPE mChb-werks,
          lgort TYPE mChb-lgort,
          charg TYPE mChb-charg,
          clabs TYPE mChb-clabs,
          lwedt TYPE mChA-lwedt,
        END OF ty_mchb.
  DATA lt_mchb TYPE TABLE OF TY_mchb.
  DATA ls_mchb TYPE          TY_mchb.

    gs_righ-zpsjhitme = <fs_head>-zpsjhitme.  "配送计划行
    gs_righ-lgort     = <fs_head>-lgort.      "发出库存

*    gs_righ-zthdnum   = ls_head-zthdnum.    "退货单号
*    gs_righ-zthditem  = ls_head-zthditem.   "退货单行

    IF <fs_head>-Xchar = 'X'.

      CLEAR:lt_mchb,ls_mchb.

  DATA lv_kosar TYPE csks-kosar.
  DATA lv_bwart TYPE mseg-bwart.

*
*ZPSJHTYPE等于C
*CSKS-KOSTL = KOSTL    “成本中心
*CSKS-KOKRS = ‘HXJT’“华星集团控制范围
*CSKS-DATAB <= 当前日期 <= CSKS-DATBI

*查询：
*CSKS-KOSAR    “成本中心类型
*如果“成本中心类型”CSKS-KOSAR等于P，
*“移动类型”BWART为201；

*如果“成本中心类型”CSKS-KOSAR不等于P

    CLEAR:lv_kosar.

      INTO LV_kosar
      FROM csks
      WHERE kostl = gs_righ-kostl
        AND kokrs = 'HXJT'
        AND ( datab <= sy-datum AND datbi >= sy-datum ).

    CASE  LV_kosar.
      WHEN 'P'.
        lv_bwart = '201'.
      WHEN 'F'.
        lv_bwart = 'Z01'.
      WHEN OTHERS.
    ENDCASE.

    ls_item-material   = gs_RIGH-matnr.           "料号
    ls_item-plant      = S_werks-low.              "工厂
    ls_item-stge_loc   = gs_RIGH-lgort.           "库存地点
