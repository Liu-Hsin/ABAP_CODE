FORM sub_get_log.
  DATA: l_ts_froml TYPE tzonref-tstampl,
        l_ts_tol   TYPE tzonref-tstampl.
  DATA: l_ts_call TYPE timestamp.

  "将日期和时间转换为 短时间戳 差8小时！
  CALL FUNCTION 'IB_CONVERT_INTO_TIMESTAMP' "FROM
    EXPORTING
      i_datlo     = p_datefr
      i_timlo     = p_timefr
    IMPORTING
      e_timestamp = g_ts_froms.

  CALL FUNCTION 'IB_CONVERT_INTO_TIMESTAMP' "TO
    EXPORTING
      i_datlo     = p_dateto
      i_timlo     = p_timeto
    IMPORTING
      e_timestamp = g_ts_tos.

  "将 短时间戳 转换为 长时间戳
  PERFORM sub_conv_ts_from_short_to_long USING g_ts_froms g_ts_tos
                                                l_ts_froml l_ts_tol.

  REFRESH: tab_log.

  SELECT call_id
         ts_call
         username
         call_state
         errortype
         calledprog
         comm_patt
         req_id
         resp_id
         op_name
         remaddr
         trc_key
    UP TO p_uprow ROWS
    INTO CORRESPONDING FIELDS OF TABLE tab_log
    FROM (log1)"srt_monilog_data
    WHERE call_id IN s_callid
      AND ts_call BETWEEN l_ts_froml AND l_ts_tol
      AND call_state IN s_state
      AND sapclient = f_mandt
      AND op_name IN s_opname
    ORDER BY ts_call DESCENDING.

  SELECT call_id
         ts_call
         username
         call_state
         errortype
         calledprog
         comm_patt
         req_id
         resp_id
         op_name
         remaddr
         trc_key
    UP TO p_uprow ROWS
    APPENDING CORRESPONDING FIELDS OF TABLE tab_log

*&---------------------------------------------------------------------*
*&      FORM  sub_filter
*&---------------------------------------------------------------------*
*       关键字过滤
*----------------------------------------------------------------------*
FORM sub_filter.
  "先过滤请求XML
  IF p_req IS NOT INITIAL.
    LOOP AT tab_log INTO rec_log.
      g_tabix = sy-tabix.

      "查找Payload表获得XML
      CLEAR: rec_trace.
      SELECT SINGLE *
        INTO rec_trace
        FROM (payload)"srt_moni_payltrc
        WHERE id = rec_log-call_id.

      "没有XML日志的删除
      IF sy-subrc <> 0.
        DELETE tab_log INDEX g_tabix.
        CONTINUE.
      ENDIF.

      "没有XML日志的删除
      IF rec_trace-soap_env_req IS INITIAL.
        DELETE tab_log INDEX g_tabix.
        CONTINUE.
      ENDIF.

      CLEAR: g_xml_content.
      CALL METHOD cl_soap_moni_helper=>convert_xstring_to_string
        EXPORTING
          xstring_in = rec_trace-soap_env_req
          charset    = 'UTF-8'
        RECEIVING
          string_out = g_xml_content.

      "不包含关键字的删除
      SEARCH g_xml_content FOR p_req.
      IF sy-subrc <> 0.
        DELETE tab_log INDEX g_tabix.
        CONTINUE.
      ENDIF.
    ENDLOOP.
  ENDIF.

  "再过滤响应XML
  IF p_resp IS NOT INITIAL.
    LOOP AT tab_log INTO rec_log.
      g_tabix = sy-tabix.

      "查找Payload表获得XML
      CLEAR: rec_trace.
      SELECT SINGLE *
        INTO rec_trace
        FROM (payload)"srt_moni_payltrc
        WHERE id = rec_log-call_id.

      IF sy-subrc <> 0.
        DELETE tab_log INDEX g_tabix.
        CONTINUE.
      ENDIF.

      IF rec_trace-soap_env_req IS INITIAL.
        DELETE tab_log INDEX g_tabix.
        CONTINUE.
      ENDIF.

      CLEAR: g_xml_content.
      CALL METHOD cl_soap_moni_helper=>convert_xstring_to_string
        EXPORTING
          xstring_in = rec_trace-soap_env_resp
          charset    = 'UTF-8'
        RECEIVING
          string_out = g_xml_content.

      SEARCH g_xml_content FOR p_resp.
      IF sy-subrc <> 0.
        DELETE tab_log INDEX g_tabix.
        CONTINUE.
      ENDIF.
    ENDLOOP.
  ENDIF.
ENDFORM. "sub_filter
