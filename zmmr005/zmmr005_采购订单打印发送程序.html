FORM select_data.
  DATA lt_out TYPE TABLE OF ty_out.

  SELECT ekko~ebeln,
         ekpo~ebelp,
         ekko~bsart,
         ekko~bstyp,
         ekko~lifnr,
         ekko~bedat,   "凭证日期
         ekko~bukrs,   "公司代码
         ekko~ekorg,   "采购组织
         ekko~ekgrp,   "采购组

         ekpo~matnr,   "物料号

         ekpo~meins,   "订单单位

         ekET~eindt,
         ekpo~adrnr

    FROM ekko
    INNER JOIN ekpo ON ekpo~ebeln = ekko~ebeln
    LEFT JOIN  eket ON eket~ebeln = ekpo~ebeln
                   AND eket~ebelp = ekpo~ebelp
                   AND eket~etenr = '0001'
    WHERE ekko~ekorg IN @s_ekorg
      AND ekko~bsart IN @s_bsart
      AND ekko~bukrs IN @s_bukrs
      AND ekko~ernam IN @s_ernam
      AND ekko~aedat IN @s_aedat
      AND ekko~lifnr IN @s_lifnr
      AND ekko~ekgrp IN @s_ekgrp
      AND ekko~bedat IN @s_bedat
      AND ekko~ebeln IN @s_ebeln
      AND ekpo~ebelp IN @s_ebelp
      AND ekpo~knttp IN @s_knttp
      AND ekpo~pstyp IN @s_PSTYP
      AND ekpo~matnr IN @s_MATNR
      AND ekpo~txz01 IN @s_TXZ01
      AND ekpo~matkl IN @s_MATKL
      AND ekpo~bednr IN @s_BEDNR
      AND ekpo~werks IN @s_WERKS
      AND ekpo~lgort IN @s_LGORT
      AND ekET~eindt IN @s_eindt
      AND ekpo~bstyp  = 'F'
      AND ekko~memory = ''.



  IF p_sen = 'X' AND gt_out[] IS NOT INITIAL.

    DATA lt_output TYPE TABLE OF ty_out.
    CLEAR:lt_output[].
    lt_output[] = gt_out[].
    SORT lt_output BY ebeln .
    DELETE ADJACENT DUPLICATES FROM lt_output COMPARING ebeln.
    CLEAR:gt_out[].

    SELECT ekko~ebeln,
       ekpo~ebelp,
       ekko~bsart,
       ekko~bstyp,
       ekko~lifnr,
       ekko~bedat,   "凭证日期
       ekko~bukrs,   "公司代码
       ekko~ekorg,   "采购组织
       ekko~ekgrp,   "采购组
       ekpo~loekz,   "删除标识
       ekpo~elikz,   "交货完成标识
       ekpo~pstyp,   "项目类别
       ekpo~matnr,   "物料号
       ekpo~txz01,   "短文本
       ekpo~menge,   "采购订单数量
       ekpo~meins,   "订单单位
       ekPO~werks,   "工厂
       ekpo~lgort,   "库存地点
       ekET~eindt,
       ekpo~adrnr
  INTO CORRESPONDING FIELDS OF TABLE @gt_out
  FROM ekko
  INNER JOIN ekpo ON ekpo~ebeln = ekko~ebeln
  LEFT JOIN  eket ON eket~ebeln = ekpo~ebeln
                 AND eket~ebelp = ekpo~ebelp
                 AND eket~etenr = '0001'
  FOR ALL ENTRIES IN @lt_output
  WHERE ekko~ebeln = @lt_output-ebeln.

  ENDIF.


*【1】未交货完成的：勾选后增加逻辑：EKPO-ELIKZ = 空
*【2】未删除的：勾选后增加逻辑：EKPO-LOEKZ = 空
*【3】未曾发送OA的：勾选后增加逻辑：排除EKKO-EBELN = ZMMPO-EBELN的条目

  CLEAR:lt_out[].
  lt_out[] = gt_out[].
  SORT lt_out BY ebeln.
  DELETE ADJACENT DUPLICATES FROM lt_out COMPARING ebeln.

  IF lt_out[] IS NOT INITIAL.
    SELECT *
      INTO CORRESPONDING FIELDS OF TABLE GT_zmmpopost
      FROM zmmpopost
      FOR ALL ENTRIES IN lt_out
      WHERE ebeln = lt_out-ebeln.
  ENDIF.

  IF p_r1 = 'X'. "未交货完成的
    DELETE gt_out WHERE elikz <> ''.
  ENDIF.

  IF p_r2 = 'X'. "未删除的
    DELETE gt_out WHERE loekz <> ''.
  ENDIF.

  IF p_r3 = 'X'. "未曾发送OA的
    LOOP AT GT_zmmpopost INTO GS_zmmpopost.
      DELETE gt_out WHERE ebeln = GS_zmmpopost-ebeln.
    ENDLOOP.
  ENDIF.

    SELECT partner name_org1
      INTO CORRESPONDING FIELDS OF TABLE gt_but000

    INTO CORRESPONDING FIELDS OF TABLE GT_t001l

    INTO CORRESPONDING FIELDS OF TABLE GT_t001

    INTO CORRESPONDING FIELDS OF TABLE GT_t024
    FROM t024.

  CLEAR:gt_t024e[].
  SELECT *
    INTO CORRESPONDING FIELDS OF TABLE GT_t024e
    FROM t024e.

  CLEAR:gt_t161t.
  SELECT *
    INTO CORRESPONDING FIELDS OF TABLE gt_t161t
    FROM t161t
    WHERE spras = sy-langu.

  CLEAR:gt_t163y.

    INTO CORRESPONDING FIELDS OF TABLE gt_t163y

FORM process_data.

  SORT gt_mara   BY matnr.
  SORT gt_t001w  BY werks.
  SORT gt_t001L  BY werks lgort.
  SORT gt_t001   BY bukrs.
  SORT gt_t024   BY ekgrp.
  SORT gt_t024E  BY ekorg.
  SORT GT_t006a  BY msehi.
  SORT GT_but000 BY partner.
  SORT GT_t161t  BY bsart bstyp.

  SORT GT_t163y BY pstyp.

  SORT GT_zmmpopost BY ebeln.

*取抬头文本
  DATA lt_out TYPE TABLE OF ty_out.
  DATA ls_out TYPE          ty_out.
  CLEAR:lt_out[].
  lt_out[] = gt_out[].
  SORT lt_out BY ebeln.
  DELETE ADJACENT DUPLICATES FROM lt_out COMPARING ebeln.
  LOOP AT lt_out ASSIGNING FIELD-SYMBOL(<ls_out>).

*取长文本
    PERFORM GET_zekkotext USING <ls_out>-ebeln
                       CHANGING <ls_out>-zekkotext.
  ENDLOOP.


  SORT lt_out BY ebeln.


  LOOP AT gt_out ASSIGNING <fs_out>.

*长文本
    CLEAR:ls_out.
    READ TABLE lt_out INTO ls_out WITH KEY ebeln = <fs_out>-ebeln BINARY SEARCH.
    <fs_out>-zekkotext = ls_out-zekkotext.

*采购订单凭证类型描述
    PERFORM get_batxt USING <fs_out>-bsart
                            <fs_out>-bstyp
                   CHANGING <fs_out>-batxt.

*供应商名称
    PERFORM GET_name_org1 USING <fs_out>-lifnr
                       CHANGING <fs_out>-name_org1.

*单位描述
    PERFORM GET_mseh6 USING <fs_out>-meins
                   CHANGING <fs_out>-mseh6.

*公司描述
    PERFORM GET_butxt USING <fs_out>-bukrs
                   CHANGING <fs_out>-butxt.

*采购组织描述
    PERFORM GET_ekotx USING <fs_out>-ekorg
                   CHANGING <fs_out>-ekotx.

*采购组描述
    PERFORM GET_eknam USING <fs_out>-ekgrp
                   CHANGING <fs_out>-eknam.
*工厂描述
    PERFORM get_name1 USING <fs_out>-werks
                   CHANGING <fs_out>-name1.

*库存地点描述
    PERFORM GET_lgobe USING <fs_out>-lgort
                            <fs_out>-werks
                   CHANGING <fs_out>-lgobe.


*
    CASE <fs_out>-loekz.
      WHEN 'X'.
        <fs_out>-loekT = '冻结'.
      WHEN 'L'.
        <fs_out>-loekT = '删除'.
      WHEN OTHERS.
        <fs_out>-loekT = '未删除'.
    ENDCASE.

    CASE <fs_out>-elikz.
      WHEN 'X'.
        <fs_out>-elikT = '交货完成或订单关闭'.
      WHEN OTHERS.
        <fs_out>-elikT = '未完成'.
    ENDCASE.

    CLEAR:GS_t163y.
    READ TABLE GT_t163y INTO GS_t163y WITH KEY pstyp = <fs_out>-pstyp BINARY SEARCH.
    IF sy-subrc = 0.
      <fs_out>-pstyT = GS_t163y-ptext.
    ENDIF.


*发送类型
    CLEAR:GS_zmmpopost.
    READ TABLE GT_zmmpopost INTO GS_zmmpopost
         WITH KEY ebeln = <fs_out>-ebeln BINARY SEARCH.
    IF sy-subrc = 0.
      <fs_out>-ztype = '2'. "2 订单变更
      <fs_out>-ztypT = '订单变更'. "2

      <fs_out>-zbname    = GS_zmmpopost-zbname.
      <fs_out>-zlastdate = GS_zmmpopost-zlastdate.
      <fs_out>-zlasttime = GS_zmmpopost-zlasttime.

    ELSE.
      <fs_out>-ztype = '1'. "1 首次发送
      <fs_out>-ztypT = '首次发送'.
    ENDIF.

  ENDLOOP.

ENDFORM.
