FORM frm_excel_to_dyntable TABLES pt_excel TYPE tt_excel
                             USING VALUE(pv_type) TYPE tabname.
*创建动态表结构
  CREATE DATA dyn_table TYPE TABLE OF (pv_type).
*创建动态内表
  ASSIGN dyn_table->* TO <dyn_table>.
*创建动态工作区结构
  CREATE DATA dyn_wa LIKE LINE OF <dyn_table>.
*创建动态工作区
  ASSIGN dyn_wa->* TO <dyn_wa>.

  FIELD-SYMBOLS: <l_feld>.
  DATA: lw_excel TYPE alsmex_tabline.
  DATA: lv_col        TYPE i.          "列号

  LOOP AT pt_excel INTO lw_excel.
    " 定位列
    CLEAR: lv_col.
    lv_col = lw_excel-col .
    ASSIGN COMPONENT lv_col OF STRUCTURE <dyn_wa> TO <l_feld> .
    IF <l_feld> IS ASSIGNED.
      <l_feld> = lw_excel-value .
    ENDIF.
    AT END OF row .
      APPEND <dyn_wa>  TO <dyn_table>.
      CLEAR: <dyn_wa>.
    ENDAT.
    CLEAR: lw_excel.
  ENDLOOP.
ENDFORM. " FRM_EXCEL_TO_DYNTABLE
