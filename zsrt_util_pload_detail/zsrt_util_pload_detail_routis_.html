        ELSE.
          SPLIT ls_part_property-tag_value AT '/' INTO p_mime_type p_mime_subtype.
        ENDIF.
      ELSEIF ls_part_property-tag_value = 'content-id'.     "#EC NOTEXT
        l_item_index = l_item_index + 1.

        ENDIF.

      IF ls_part_property-tag_name <> '---MIME_HEADER'.     "#EC NOTEXT

*&---------------------------------------------------------------------*
*&      Form  update_property
*&---------------------------------------------------------------------*
FORM update_property USING p_row_no TYPE i.

  DATA: l_ignore_level  TYPE i,
        l_next_line     TYPE i,
        l_expand_line   TYPE i,
        l_delete_line   TYPE i,
        l_ignore        TYPE xsdboolean,
        l_no_last_line  TYPE xsdboolean,
        ls_prop         TYPE ty_property,
        ls_org_prop     TYPE ty_property,
        ls_current_prop TYPE ty_property,
        ls_last_prop    TYPE ty_property.

  CLEAR m_update_required.
  READ TABLE mt_property INDEX p_row_no INTO ls_current_prop.
  IF sy-subrc <> 0. EXIT. ENDIF.
  ls_org_prop = ls_current_prop.

* Expand Node or Display Attachments
  IF ls_current_prop-tag_type <> cl_soap_xml_parser=>co_tag_type_node_only.
    READ TABLE mt_property INTO ls_prop INDEX 1.
    IF ls_prop-tag_name = '-MULTIPART' AND ls_current_prop-tag_name = '---PART_CONTENT'.
      PERFORM show_attachment USING p_row_no.
      EXIT.
    ENDIF.
    EXIT.
  ENDIF.

* Get location for insert
  m_update_required = 'X'.
  l_next_line = sy-tabix + 1.
  READ TABLE mt_property INTO ls_last_prop INDEX l_next_line.
  IF sy-subrc <> 0.
    l_no_last_line = 'X'.
    READ TABLE mt_prop_expand INTO ls_last_prop INDEX m_total_lines.
  ELSE.
    IF ls_last_prop-tag_type = cl_soap_xml_parser=>co_tag_type_attribute.
      DO.
        l_next_line = l_next_line + 1.
        READ TABLE mt_property INTO ls_last_prop INDEX l_next_line.
        IF sy-subrc <> 0.
          l_no_last_line = 'X'.
          READ TABLE mt_prop_expand INTO ls_last_prop INDEX m_total_lines.
          EXIT.
        ELSE.
          IF ls_last_prop-tag_type <> cl_soap_xml_parser=>co_tag_type_attribute.
            EXIT.
          ENDIF.
          ls_current_prop = ls_last_prop.
        ENDIF.
      ENDDO.
    ENDIF.
  ENDIF.

* Expand
  IF ls_org_prop-expanded IS INITIAL.
    PERFORM set_icon_collapse USING ls_org_prop.
    MODIFY mt_property FROM ls_org_prop INDEX p_row_no.
    MODIFY mt_prop_expand FROM ls_org_prop INDEX ls_org_prop-line_no.

* Do nothing if Empty Node
    IF ls_org_prop-line_no = ls_org_prop-last_line. EXIT. ENDIF.

* Expand until next entry
    l_expand_line = ls_current_prop-line_no + 1.
    LOOP AT mt_prop_expand FROM l_expand_line INTO ls_prop.
      IF ls_prop-line_no   > ls_last_prop-line_no OR
         ( ls_prop-line_no = ls_last_prop-line_no AND l_no_last_line IS INITIAL ).
        EXIT.
      ENDIF.
      IF l_ignore IS INITIAL.
        IF ls_prop-tag_level = ls_current_prop-tag_level AND
             ls_prop-tag_type  = cl_soap_xml_parser=>co_tag_type_attribute.
          CONTINUE.
        ELSE.
          INSERT ls_prop INTO mt_property INDEX l_next_line.
          l_next_line = l_next_line + 1.
        ENDIF.
      ELSE.
        IF ls_prop-tag_level > l_ignore_level.
          CONTINUE.
        ELSE.
          INSERT ls_prop INTO mt_property INDEX l_next_line.
          l_next_line = l_next_line + 1.
          IF ls_prop-tag_level <= l_ignore_level AND
             ls_prop-tag_type <> cl_soap_xml_parser=>co_tag_type_attribute.
            CLEAR: l_ignore, l_ignore_level.
          ENDIF.
        ENDIF.
      ENDIF.
      IF ls_prop-expanded IS INITIAL AND
         ls_prop-tag_type = cl_soap_xml_parser=>co_tag_type_node_only.
        l_ignore = 'X'.
        l_ignore_level = ls_prop-tag_level.
      ENDIF.
    ENDLOOP.

* Collapse
  ELSE.
    PERFORM set_icon_expand USING ls_org_prop.
    MODIFY mt_property FROM ls_org_prop INDEX p_row_no.
    MODIFY mt_prop_expand FROM ls_org_prop INDEX ls_org_prop-line_no.

* Do nothing if Empty Node
    IF ls_org_prop-line_no = ls_org_prop-last_line. EXIT. ENDIF.

* Collapse until next level
    LOOP AT mt_property FROM l_next_line INTO ls_prop.
      l_delete_line = sy-tabix.
      IF ls_prop-line_no > ls_org_prop-last_line. EXIT. ENDIF.
      IF ls_prop-tag_level = ls_org_prop-tag_level AND
         ls_prop-tag_type = cl_soap_xml_parser=>co_tag_type_attribute.
        CONTINUE.
      ENDIF.
      DELETE mt_property INDEX l_delete_line.
    ENDLOOP.

  ENDIF.

ENDFORM. " update_property
