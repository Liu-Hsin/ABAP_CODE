FORM print_data .
  DATA: ls_head TYPE zfis003,
        ls_item TYPE zfis003,
        lt_item LIKE TABLE OF ls_item.

  DATA: lv_tdmhab TYPE  tdmhab,
        lv_tdmsol TYPE tdmsol.

  control_parameters-no_open = 'X'.
  control_parameters-no_close = 'X'.
  control_parameters-langu = '1'.   " CHINESE LANGUAGE

  output_options-tddest   = 'LP01'.
  output_options-tdiexit = 'X'.
  output_options-tddelete = 'X'.
  output_options-tdimmed = 'X'.


  IF gt_out[] IS NOT INITIAL.

    READ TABLE gt_out INTO gs_out WITH KEY checkbox = 'X'.
    IF sy-subrc <> 0.
      MESSAGE i888(sabapdocu) WITH '清勾选打印的凭证!'.
      RETURN.

    ELSE.

      PERFORM: open_form.

      LOOP AT gt_out INTO gs_out WHERE checkbox = 'X'.

        LOOP AT gt_tab INTO gs_tab WHERE belnr = gs_out-belnr.

          "总计:借方金额
          gs_out-tdmhab = gs_out-tdmhab + gs_tab-vdval.
          "总计:贷方金额
          gs_out-tdmsol = gs_out-tdmsol + gs_tab-vcval.

          MOVE-CORRESPONDING gs_tab TO ls_item.

          APPEND ls_item TO lt_item.
          CLEAR: ls_item,gs_tab.

        ENDLOOP.
        IF p_box2 = 'X'.
          " 汇总数据。
          PERFORM: tabulate_data TABLES lt_item CHANGING lt_item. " 汇总数据 同科目同借贷
        ENDIF.

        " 金额转大写
        call function 'ZFI_FM001'
          EXPORTING
            i_dmbtr = gs_out-tdmsol
          IMPORTING
            o_dmbtr = gs_out-zjedx.

        MOVE-CORRESPONDING gs_out TO ls_head.

        CLEAR: gs_tab.
        " 传入数据至打印
        PERFORM print_sf_data TABLES lt_item USING ls_head .

        CLEAR: lt_item[],ls_item,ls_head,gs_out.

      ENDLOOP.

      PERFORM: close_form.
    ENDIF.

  ENDIF.
ENDFORM.

*& FORM TABULATE_DATA
*&---------------------------------------------------------------------*
FORM tabulate_data TABLES p_lt_item STRUCTURE zfis003 CHANGING lt_item.

  DATA: ls_item TYPE zfis003,
        ls_tmp  TYPE zfis003,
        lt_tmp  LIKE TABLE OF ls_tmp.


  LOOP AT p_lt_item INTO ls_item

    GROUP BY ( hkont = ls_item-hkont xshkz = ls_item-xshkz ).
    MOVE-CORRESPONDING ls_item TO ls_tmp.

    ls_tmp-vdval = REDUCE #( INIT sum_vd TYPE fc_vdval
                              FOR wa_reduce IN GROUP ls_item
                              NEXT sum_vd = sum_vd + wa_reduce-vdval ).

    ls_tmp-vcval = REDUCE #( INIT sum_vc TYPE fc_vcval
                              FOR wa_reduce IN GROUP ls_item
                              NEXT sum_vc = sum_vc + wa_reduce-vcval ).
    APPEND ls_tmp TO lt_tmp.
    CLEAR: ls_tmp,ls_item.
  ENDLOOP.

  p_lt_item[] = lt_tmp[].

ENDFORM.
