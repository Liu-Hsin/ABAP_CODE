FORM select_data_r1.

  SELECT aUFK~werks afko~rsnum
         afko~plnbez AS matnr
         resb~matnr  AS idnrk
         afko~aufnr
         reSb~enmng
         reSb~bdmng
         reSb~meins
     INTO CORRESPONDING FIELDS OF TABLE gt_out
     FROM afko
     INNER JOIN aufk ON aufk~aufnr = afko~aufnr
     INNER JOIN resB ON resb~rsnum = afko~rsnum
     WHERE afko~aufnr IN s_aufnr
       AND resb~xloek = ''
       AND resb~dumps = ''
       AND resb~shkzg = 'H'.


  DATA lt_out TYPE TABLE OF ty_out.
  CLEAR:lt_out.
  lt_out[] = gt_out[].
  SORT lt_out BY matnr.
  DELETE ADJACENT DUPLICATES FROM lt_out COMPARING matnr.
  IF lt_out[] IS NOT INITIAL.
    SELECT *
      INTO CORRESPONDING FIELDS OF TABLE gt_makt
      FROM makt
      FOR ALL ENTRIES IN lt_out
      WHERE matnr = lt_out-matnr
        AND spras = sy-langu.
  ENDIF.


  CLEAR:lt_out.
  lt_out[] = gt_out[].
  SORT lt_out BY idnrk.
  DELETE ADJACENT DUPLICATES FROM lt_out COMPARING idnrk.
  IF lt_out[] IS NOT INITIAL.
    SELECT *
      APPENDING CORRESPONDING FIELDS OF TABLE gt_makt
      FROM makt
      FOR ALL ENTRIES IN lt_out
      WHERE matnr = lt_out-idnrk
        AND spras = sy-langu.


    SELECT matnr werks lgort labst
      INTO CORRESPONDING FIELDS OF TABLE gt_mard
      FROM mard
      FOR ALL ENTRIES IN lt_out
      WHERE matnr = lt_out-idnrk
        AND werks = p_werks.

    SELECT marc~matnr
           marc~werks
           mara~matkl
           mara~mtart
           mara~bismt
           marc~beskz
      INTO CORRESPONDING FIELDS OF TABLE gt_marc
      FROM marc
      INNER JOIN mara ON mara~matnr = marc~matnr
      FOR ALL ENTRIES IN lt_out
      WHERE marc~werks = p_werks
        AND marc~matnr = lt_out-idnrk.

** Modified by Vincent, add the bismt for matnr **

     SELECT mara~matnr  mara~bismt AS bismt1
      INTO CORRESPONDING FIELDS OF TABLE gt_mara
      FROM  mara
      FOR ALL ENTRIES IN lt_out
      WHERE mara~matnr = lt_out-matnr.


  ENDIF.

  SELECT *
    INTO TABLE gt_t023t
    FROM t023t
    WHERE spras = sy-langu.


ENDFORM.

*&---------------------------------------------------------------------*
FORM process_data_r1.

  DATA lt_mard TYPE TABLE OF ty_mard.
  DATA ls_mard TYPE          ty_mard.
  CLEAR:lt_mard,ls_mard.
  lt_mard[] = gt_mard[].
  CLEAR:gt_mard.
  LOOP AT lt_mard INTO ls_mard.
    CLEAR:ls_mard-lgort.
    COLLECT ls_mard INTO gt_mard.
  ENDLOOP.


  SORT gt_makt  BY matnr.
  SORT gt_marc  BY matnr.
  SORT gt_mara  BY matnr.
  SORT gt_mard  BY matnr.
  SORT gt_t023t BY matkl.


*汇总
  DATA lt_out TYPE TABLE OF ty_out.
  CLEAR:lt_out.
  lt_out[] = gt_out[].
  CLEAR:GT_out[].
  LOOP AT lt_out INTO gs_out.
    COLLECT gs_out INTO gt_out.
  ENDLOOP.




  LOOP AT gt_out ASSIGNING <fs_out>.
    CLEAR:gs_makt.
    READ TABLE gt_makt INTO gs_makt WITH KEY matnr = <fs_out>-matnr BINARY SEARCH.
    <fs_out>-maktx = gs_makt-maktx.

    CLEAR:gs_makt.
    READ TABLE gt_makt INTO gs_makt WITH KEY matnr = <fs_out>-idnrk BINARY SEARCH.
    <fs_out>-IDNRx = gs_makt-maktx.

    CLEAR:gs_marc.
    READ TABLE gt_marc INTO gs_marc WITH KEY matnr = <fs_out>-idnrk BINARY SEARCH.
    <fs_out>-matkl = gs_marc-matkl.
    <fs_out>-beskz = gs_marc-beskz.
*&---------------------------------------------------------------------*
*&                 程序修改 *&  增加图号
    <fs_out>-bismt = gs_marc-bismt.
*&---------------------------------------------------------------------*


CLEAR:gs_mara.
    READ TABLE gt_mara INTO gs_mara WITH KEY matnr = <fs_out>-matnr BINARY SEARCH.
     <fs_out>-bismt1 = gs_mara-bismt1.



*opnres 未清预留量
    <fs_out>-opnres = <fs_out>-bdmng - <fs_out>-enmng.


*otrres  其他订单预留量
    PERFORM GET_otrres USING <fs_out>-idnrk
                             <Fs_OUT>-aufnr
                    CHANGING <fs_out>-otrres.


*avlstk 可用库存量
    CLEAR:gs_mard.
    READ TABLE gt_mard INTO gs_mard WITH KEY matnr = <fs_out>-idnrk BINARY SEARCH.
    <fs_out>-avlstk = gs_mard-labst.


*opnpo 未清采购订单
    IF <fs_out>-beskz = 'F'.
      PERFORM get_opnpo USING <fs_out>-idnrk
                     CHANGING <fs_out>-opnpo.


    ELSEIF <fs_out>-beskz = 'E'.
*opnwo 未清生产订单
      PERFORM get_opnwo USING <fs_out>-idnrk
                     CHANGING <fs_out>-opnwo.
    ENDIF.


*shortage1 短缺量（仅库存）
*该组件的可用库存量（AVLSTK）-未清预留量（OPNRES）-其他订单预留量（OTRRES），
*如果为负数，则显示红色，如果为正数，则显示绿色。
    <fs_out>-shortage1 = <fs_out>-avlstk
                       - <fs_out>-opnres
                       - <fs_out>-otrres.


*shortage2 短缺量（含订单）
*该组件的 可用库存量（AVLSTK）+ 未清采购订单（OPNPO）+ 未清生产订单（OPNWO）
*- 未清预留量（OPNRES）- 其他订单预留量（OTRRES），
*如果为负数，则显示红色，如果为正数，则显示绿色。
    <fs_out>-shortage2 = <fs_out>-avlstk
                       + <fs_out>-opnpo
                       + <fs_out>-opnwo
                       - <fs_out>-otrres
                       - <fs_out>-opnres.

    CLEAR:gs_t023t.
    READ TABLE gt_t023t INTO gs_t023t WITH KEY matkl = <fs_out>-matkl BINARY SEARCH.
    <fs_out>-wgbez = gs_t023t-wgbez.

  ENDLOOP.



ENDFORM.

FORM GET_otrres USING UV_idnrk
                          uv_aufnr
                 CHANGING CV_otrres.
  DATA LV_bdmng TYPE resb-bdmng.
  DATA LV_enmng TYPE resb-enmng.
  CLEAR:LV_enmng,LV_bdmng.

  SELECT  SUM( resb~bdmng )
          SUM( resb~enmng )
    INTO ( LV_bdmng,lv_enmng )
    FROM resb
    WHERE bdart = 'AR'
      AND xwaok = 'X'
      AND xloek = ''
      AND dumps = ''
      AND shkzg = 'H'
      AND matnr = uv_idnrk
      AND aufnr <> uv_aufnr.

  CV_otrres = LV_bdmng - LV_enmng.

ENDFORM.

*&---------------------------------------------------------------------*

*&---------------------------------------------------------------------*
FORM get_opnpo USING UV_idnrk
                CHANGING CV_opnpo.

  DATA lv_menge_ekbe TYPE ekbe-menge.
  DATA lv_menge_ekPO TYPE ekpo-menge.
  CLEAR:lv_menge_ekbe,LV_menge_ekPO.

  SELECT SUM( ekbe~menge  *
    ( CASE ekbe~shkzg
        WHEN 'H' THEN ( -1 )  "贷方
        WHEN 'S' THEN ( 1 )   "借方
      END ) )
      INTO @lv_menge_ekbe
      FROM ekbe
      INNER JOIN ekpo ON ekpo~ebeln = ekbe~ebeln
                     AND ekpo~ebelp = ekbe~ebelp
      WHERE ekpo~matnr = @uv_idnrk
        AND ekpo~loekz = ''
        AND ekpo~elikz = ''.


  SELECT SUM( ekPO~menge )
      INTO @lv_menge_ekPO
      FROM ekPO
      WHERE matnr = @uv_idnrk
        AND loekz = ''
        AND elikz = ''.

  CV_opnpo = lv_menge_ekpo - lv_menge_ekbe.

ENDFORM.

FORM get_opnwo USING UV_idnrk
            CHANGING CV_opnWO.

*11.  未清生产订单（OPNWO）：如果该组件的采购类型为E，
*则根据每个组件料号查询表AFKO和AFPO中生产订单状态包含REL（TJ02T-ISTAT=I0002），
*不含‘标记’/DLV/TECO/‘结算’（TJ02T-ISTAT=I0076/I0012/I0045/I0046）
*的生产订单（AFKO-AUFNR-->AUFK-OBJNR --> JEST-STAT
*TJ02T-ISTAT），订单类型（AFPO-DAUAT）包括“Z001”的生产订单，
*并汇总该组件所有订单中的未收货量（AFPO-PSMNG - AFPO-WEMNG）作为该组件的未清生产订单量（OPNWO）。

  SELECT afko~aufnr,
         afko~plnbez,
         aufk~werks,
         aufk~objnr,
         aFPO~psmng,
         aFPO~wemng
    INTO TABLE @DATA(Lt_afko)
    FROM afko
    INNER JOIN aufk ON aufk~aufnr = afko~aufnr
    INNER JOIN jest ON jest~objnr = aufk~objnr
    INNER JOIN afpo ON afpo~aufnr = aufk~aufnr
    WHERE plnbez = @uv_idnrk
      AND werks  = @p_werks
      AND ( jest~stat = 'I0002' AND jest~inact = '' ) " AND jest~stat = 'I0076' AND jest~inact = 'X' )
      AND afpo~dauat = 'Z001'.

*存在  REL 和删除同时存在的订单
  DATA LV_objnr TYPE jest-objnr.

  LOOP AT lt_afko INTO DATA(ls_afko).
    CLEAR:LV_objnr.
    SELECT SINGLE objnr
      FROM jest
      INTO lv_objnr
      WHERE objnr = ls_afko-objnr
        AND stat = 'I0076'
        AND inact = ''.

    IF sy-subrc = 0.
      CONTINUE.
    ENDIF.

    CV_opnWO = CV_opnWO +  ls_afko-psmng - ls_afko-wemng.
  ENDLOOP.

ENDFORM.
