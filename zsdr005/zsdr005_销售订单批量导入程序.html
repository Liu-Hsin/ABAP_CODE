FORM read_excel.

  DATA lv_filename TYPE rlgrap-filename.
  DATA :lv_index TYPE i.
  FIELD-SYMBOLS : <fs> TYPE any.
  CLEAR lv_filename.
  lv_filename = p_file.

  CALL FUNCTION 'ALSM_EXCEL_TO_INTERNAL_TABLE'
    EXPORTING
      filename                = lv_filename
      i_begin_col             = 1
      i_begin_row             = 3
      i_end_col               = 256
      i_end_row               = 65535
    TABLES
      intern                  = gt_excel
    EXCEPTIONS
      inconsistent_parameters = 1
      upload_ole              = 2
      OTHERS                  = 3.

  IF sy-subrc <> 0.
    MESSAGE ID sy-msgid TYPE sy-msgty NUMBER sy-msgno
            WITH sy-msgv1 sy-msgv2 sy-msgv3 sy-msgv4.
  ENDIF.

  SORT gt_excel BY row col.

  CLEAR:gt_out.

  LOOP AT gt_excel INTO gs_excel.

    MOVE gs_excel-col TO lv_index.
    ASSIGN COMPONENT lv_index OF STRUCTURE gs_OUT TO <fs>.
    MOVE gs_excel-value TO <fs>.

    AT END OF row.
      APPEND gs_OUT TO gt_OUT.
      CLEAR gs_OUT.
    ENDAT.

  ENDLOOP.

ENDFORM.

FORM process_data.


  DATA: LS_fieldstyle TYPE lvc_s_styl.

  DATA lv_zsn TYPE sy-tabix.
  CLEAR:lv_zsn.

  LOOP AT gt_out ASSIGNING <fs_out>.

    ADD 1 TO lv_zsn.

    <fs_out>-zsn = lv_zsn.

    <fs_out>-kunnr = '0000800000'.

    SELECT SINGLE matnr
      INTO <fs_out>-matnr
      FROM knmt
      WHERE kdmat = <fs_out>-kdmat
        AND vkorg = '1001'
        AND vtweg = '00'
        AND kunnr = <fs_out>-kunnr.

    IF <fs_out>-matnr IS INITIAL.
      <fs_out>-message = '未维护客户物料数据'.
      <fs_out>-status  = icon_red_light.
      CONTINUE.
    ELSE.
      SELECT SINGLE maktx
        INTO <fs_out>-maktx
        FROM makt
        WHERE matnr = <fs_out>-matnr
          AND spras = sy-langu.
    ENDIF.

    CLEAR:LS_fieldstyle.
    IF <fs_out>-status = icon_red_light.
      LS_fieldstyle-fieldname = 'ZBOX'.
      LS_fieldstyle-style     = cl_gui_alv_grid=>mc_style_disabled.
      APPEND LS_fieldstyle TO <FS_out>-field.
    ENDIF.

  ENDLOOP.
