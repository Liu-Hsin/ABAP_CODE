FORM save.
  DATA lv_answer TYPE c.
  CLEAR:lv_answer.

  PERFORM confirm
              USING
                 '保存数据'
                 '请确认是否执行'
              CHANGING
                 lv_answer.

  CHECK  lv_answer = 1.






* ZREL = 'X'  释放
* ZREL = ''   取消释放
*ZMMLLPSJHD-ZSFZ  最近一次释放者
*ZMMLLPSJHD-ZSFDATE 最近一次释放日期
*ZMMLLPSJHD-ZSFTIME 最近一次释放时间
  DATA lv_zrel TYPE zrel.
  CLEAR:lv_zrel.

  CASE 'X'.
    WHEN p_r1. "释放
      lv_zrel = 'X'.
    WHEN p_r2. "取消释放
      lv_zrel = ''.
  ENDCASE.

  LOOP AT gt_out INTO gs_out.
    UPDATE zmmllpsjhd  SET zrel = lv_zrel
                          zsfz = sy-uname
                          zsfdate = sy-datum
                          zsftime = sy-uzeit
                     WHERE zlldnum = gs_out-zlldnum
                       AND zllditem = gs_out-zllditem.
    COMMIT WORK.
  ENDLOOP.

  MESSAGE '数据保存成功！' TYPE 'S'.

ENDFORM.
