          INTO p_msg.
        RETURN.
      ENDIF.

    WHEN '2'.
      SELECT SINGLE werks
        FROM vbap INTO lv_werks
        WHERE vbeln = p_vbeln
          AND posnr = p_posnr
          AND matnr = p_matnr.

      SELECT SINGLE matnr
        FROM kdst INTO lv_matnr
        WHERE matnr = p_matnr
          AND werks = lv_werks
          AND stlan = p_stlan
          AND vbeln = p_vbeln
          AND vbpos = p_posnr.
      IF sy-subrc = 0.
        CONCATENATE
          '抬头物料' p_matnr
          '工厂' lv_werks
          'BOM用途' p_stlan '的BOM主数据已存在'
          INTO p_msg.
        RETURN.
      ENDIF.

    WHEN OTHERS.
  ENDCASE.



ENDFORM.
