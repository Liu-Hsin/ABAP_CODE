FORM frm_convert_data .

  FIELD-SYMBOLS: <lfs_list>  TYPE gty_list,
                 <lfs_field>.

*** 权限检查
  LOOP AT gt_string.
    MOVE-CORRESPONDING gt_string TO gt_list.
    APPEND gt_list.
  ENDLOOP.
  DATA(copytab) = gt_list[].
  SORT copytab BY werks.
  DELETE ADJACENT DUPLICATES FROM copytab COMPARING werks.
  DELETE copytab WHERE werks IS INITIAL.

  IF copytab IS NOT INITIAL.
    SELECT werks
      FROM t001w
      INTO TABLE @DATA(lt_werks)
      FOR ALL ENTRIES IN @copytab
      WHERE werks = @copytab-werks.
  ENDIF.

  LOOP AT copytab ASSIGNING FIELD-SYMBOL(<fs_auth>).
    READ TABLE lt_werks TRANSPORTING NO FIELDS WITH KEY werks = <fs_auth>-werks.
    IF sy-subrc = 0.
      AUTHORITY-CHECK OBJECT 'C_AFKO_AWA'
       ID 'ACTVT' DUMMY
       ID 'AUTYP' DUMMY
       ID 'AUFART' DUMMY
       ID 'WERKS' FIELD <fs_auth>-werks.
      IF sy-subrc <> 0.
*        <FS_AUTH>-ZZMSG = |[您没有工厂{ <FS_AUTH>-WERKS }的权限]|.
        <fs_auth>-icon  = gc_red.
      ENDIF.
    ELSE.
      <fs_auth>-zzmsg = |[工厂{ <fs_auth>-werks }不存在]|.
      <fs_auth>-icon  = gc_red.
    ENDIF.

    IF <fs_auth>-icon  = gc_red .
      MODIFY gt_list FROM <fs_auth> TRANSPORTING icon zzmsg WHERE werks = <fs_auth>-werks.
    ENDIF.

  ENDLOOP.


  LOOP AT gt_list[] ASSIGNING <lfs_list>.

    PERFORM frm_matn1_input  CHANGING <lfs_list>-matnr.
    PERFORM frm_alpha_input  CHANGING <lfs_list>-prctr.
    PERFORM frm_cunit_input  CHANGING <lfs_list>-meins.
    PERFORM frm_cunit_input  CHANGING <lfs_list>-gewei.
    PERFORM frm_cunit_input  CHANGING <lfs_list>-bstme.
    PERFORM frm_cunit_input  CHANGING <lfs_list>-ausme.
    PERFORM frm_perkz_input  CHANGING <lfs_list>-iprkz.
*    PERFORM frm_abpsn_input  CHANGING <lfs_list>-zposid.

    <lfs_list>-bismt  = to_upper( <lfs_list>-bismt ).
*    <LFS_LIST>-ZMATNR = TO_UPPER( <LFS_LIST>-ZMATNR ).

    IF <lfs_list>-error1 =  'X'.
      <lfs_list>-zzmsg   = '数据类型错误！'.
      <lfs_list>-icon    = gc_red.
      gv_error1       =  'X'.
    ENDIF.

  ENDLOOP.

  MESSAGE s001(00) WITH |取得{ lines( gt_list[] ) }条数据| .
  RETURN.

ENDFORM.

  ENDIF.
