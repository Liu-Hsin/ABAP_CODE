*&---------------------------------------------------------------------*
FORM conversion_DATA.
  CLEAR:gt_out.

  DATA lt_data TYPE TABLE OF ty_data.
  CLEAR:lt_data.
  lt_data[] = gt_data[].
  SORT lt_data BY matnr.
  DELETE ADJACENT DUPLICATES FROM lt_data COMPARING matnr.
  IF lt_data[] IS NOT INITIAL.
    SELECT *
      INTO TABLE gt_makt
      FROM makt
      FOR ALL ENTRIES IN lt_data
      WHERE matnr = lt_data-matnr.

    SELECT werks matnr dispo
      INTO CORRESPONDING FIELDS OF TABLE gt_marc
      FROM marc
      FOR ALL ENTRIES IN lt_data
      WHERE matnr = lt_data-matnr
        AND werks = p_werks.

    SELECT *
      INTO CORRESPONDING FIELDS OF TABLE gt_mast
      FROM mast
      FOR ALL ENTRIES IN lt_data
      WHERE matnr = lt_data-matnr
        AND werks = p_werks.
  ENDIF.

*matnr  物料
*datum  日期
*menge  数量

  SORT gt_makt BY matnr.
  SORT gt_marc BY matnr.
  SORT gt_mast BY matnr.

  LOOP AT gt_data INTO gs_data.
    CLEAR:gs_out.
    gs_out-werks = p_werks.
    gs_out-matnr = gs_data-matnr.
    gs_out-datum = gs_data-datum.
    gs_out-menge = gs_data-menge.

    CLEAR:gs_makt.
    READ TABLE gt_makt INTO gs_makt WITH KEY matnr = gs_data-matnr BINARY SEARCH.
    IF sy-subrc <> 0.
      gs_out-zstat   = 'E'.
      gs_out-message = '该物料在系统中不存在，请检查！'.
    ELSE.
      gs_out-maktx = gs_makt-maktx.
    ENDIF.


    IF gs_out-zstat <> 'E'.
      CLEAR:gs_marc.
      READ TABLE gt_marc INTO gs_marc WITH KEY matnr = gs_data-matnr BINARY SEARCH.
      IF sy-subrc <> 0.
        gs_out-zstat   = 'E'.
        gs_out-message = '该物料在工厂' && p_werks && '下不存在，请检查！'.
      ELSE.
        gs_out-dispo = gs_marc-dispo.
      ENDIF.
    ENDIF.

*
    IF gs_out-zstat <> 'E'.
      CLEAR:GS_mast.
      READ TABLE GT_mast INTO GS_mast WITH KEY matnr = gs_data-matnr BINARY SEARCH.
      IF sy-subrc <> 0.
        gs_out-zstat   = 'E'.
        gs_out-message = '该物料在工厂' && p_werks && '下不存在BOM，请检查！！'.
      ENDIF.
    ENDIF.

    IF gs_out-zstat <> 'E'.
      gs_out-zstat   = 'S'.
      gs_out-message = '成功！'.
    ENDIF.

    APPEND gs_out TO gt_out.
  ENDLOOP.


ENDFORM.
