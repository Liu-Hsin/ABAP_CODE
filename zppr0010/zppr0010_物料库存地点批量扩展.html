DATA: g_repid LIKE sy-repid,
      wa_fieldcat TYPE lvc_s_fcat,             "定义ALV输出列：列头描述，列对应的内表字段及类型等
      gt_fieldcat TYPE lvc_t_fcat,
      gs_layout   TYPE lvc_s_layo,             "定义ALV布局参数
      gt_exclude  TYPE ui_functions,
      wa_exclude  TYPE ui_func,
      wa_events   TYPE LINE OF slis_t_event,   "定义ALV事件
      it_events   TYPE slis_t_event,
      wa_sort     TYPE lvc_s_sort,             "定义ALV排序字段
      it_sort     TYPE lvc_t_sort.
DATA:
  BEGIN OF it_output OCCURS 0 ,
    "
    matnr    LIKE mard-matnr,
    werks    LIKE mard-werks,
    lgort    LIKE mard-lgort,
    light    LIKE icon-id,
    msg(200),    "

  END OF it_output.


*    light = '@5B@'.   "正确，绿灯

DEFINE m_alv_fieldcat.
  CLEAR: wa_fieldcat.

  wa_fieldcat-fieldname     = &1.
  wa_fieldcat-scrtext_l     = &2.
  wa_fieldcat-outputlen     = &3.

  APPEND wa_fieldcat TO  gt_fieldcat.

END-OF-DEFINITION.
SELECT-OPTIONS:s_matnr FOR mara-matnr . "OBLIGATORY .
SELECT-OPTIONS:s_mtart FOR mara-mtart .
SELECT-OPTIONS:s_matkl FOR mara-matkl .
SELECT-OPTIONS:s_prdha FOR mara-prdha.
SELECT-OPTIONS: d_ersda FOR mara-ersda .
SELECT-OPTIONS:s_werks FOR mard-werks DEFAULT '1000' OBLIGATORY NO INTERVALS NO-EXTENSION .
SELECT-OPTIONS:s_lgort FOR mard-lgort OBLIGATORY .
PARAMETERS: mode LIKE apqi-putactive DEFAULT 'N' .
*PARAMETERS: lv_log TYPE mdat1-kz_aend DEFAULT ''.

START-OF-SELECTION.
  BREAK ibmchensy.

  DATA lv_waitb  TYPE i.
  DATA lv_waita  TYPE i.
  DATA lv_lock TYPE char1.
  DATA:lv_fval    TYPE char30.
  DATA:wa_message TYPE char200.
  SELECT * FROM marc AS a
    INNER JOIN mara AS b ON a~matnr = b~matnr
    INTO CORRESPONDING FIELDS OF TABLE t_marc
    WHERE a~matnr IN s_matnr
    AND a~werks IN s_werks
    AND ( a~pstat LIKE '%L%' OR a~pstat LIKE '%D%')
    AND b~mtart IN s_mtart
    AND (  b~ersda IN d_ersda OR b~laeda IN d_ersda )
    AND b~matkl IN s_matkl
    AND b~prdha IN s_prdha.

  SELECT * FROM t001l
     INTO CORRESPONDING FIELDS OF TABLE t_t001l
     WHERE werks IN s_werks
 AND lgort IN s_lgort.

  CHECK t_marc[] IS NOT INITIAL.

  SELECT * FROM mard
    INTO CORRESPONDING FIELDS OF TABLE t_mard
    FOR ALL ENTRIES IN t_marc
    WHERE werks = t_marc-werks
      AND matnr = t_marc-matnr
    .
  LOOP AT t_t001l .
    LOOP AT t_marc .
      READ TABLE t_mard WITH KEY matnr = t_marc-matnr werks = t_t001l-werks lgort = t_t001l-lgort.
      IF sy-subrc EQ 0 .
        it_output-matnr =  t_marc-matnr.
        it_output-werks =  t_marc-werks.
        it_output-lgort =  t_t001l-lgort.
        it_output-light = '@5A@'  .
        it_output-msg =  '库存地点之前已扩展'.
        APPEND it_output.
        CONTINUE .
      ENDIF.
      PERFORM frm_dynpro USING:
                                        'X'    'SAPMM03M'           '0105',
               ''    'BDC_CURSOR'           'MDAT1-KZ_AEND',
               ''    'BDC_OKCODE'           '/00',
                                        ' '    'RM03M-MATNR'        t_marc-matnr,
               ' '    'RM03M-WERKS'        t_marc-werks ,
                                        ' '    'RM03M-LFLAG'      '' ,
                                        ' '    'RM03M-NFLAG'         'X'.
*                                        '' 'MDAT1-KZ_AEND' lv_log.
      CONCATENATE 'RM03M-LGORT('       '01)'          INTO lv_fval.
      PERFORM frm_dynpro USING:
          'X'   'SAPMM03M'  '0195'  ,
          '' 'BDC_CURSOR'  'RM03M-LGORT(01)',
          '' 'BDC_OKCODE' '/00' ,
          '' lv_fval  t_t001l-lgort .

      PERFORM frm_dynpro USING:
         'X'   'SAPMM03M'  '0195'  ,
              '' 'BDC_CURSOR'  'RM03M-LGORT(01)',
               '' 'BDC_OKCODE'  '=BU' .

      PERFORM frm_call_tcode USING 'MMSC'.
      READ TABLE it_mesgtab WITH KEY msgid = 'M3' msgnr = '897' .
      IF sy-subrc EQ 0 .
        lv_lock = 'X' .
        lv_waitb = 99999 .
        CLEAR  lv_waita.
        WHILE ( lv_waita < lv_waitb  ).
          lv_waita = lv_waita + 1 .
        ENDWHILE .
        CLEAR  lv_waita .
        WHILE ( lv_waita < lv_waitb AND lv_lock EQ 'X'  ).
          REFRESH it_mesgtab[].
          PERFORM frm_call_tcode USING 'MMSC'.
          READ TABLE it_mesgtab WITH KEY msgid = 'M3' msgnr = '897' .
          IF sy-subrc EQ 0 .
            lv_lock = 'X'.
          ELSE.
            CLEAR lv_lock .
          ENDIF.
          lv_waita = lv_waita + 1 .
        ENDWHILE .
      ENDIF.
      it_output-matnr =  t_marc-matnr.
      it_output-werks =  t_marc-werks.
      it_output-lgort =  t_t001l-lgort.
      CLEAR it_output-msg .
      LOOP AT  it_mesgtab WHERE msgtyp = 'E' .
        CONCATENATE  it_mesgtab-msgv1 it_mesgtab-msgv2  it_mesgtab-msgv3 it_mesgtab-msgv4 it_output-msg INTO it_output-msg .
        it_output-light = '@5C@'  .
      ENDLOOP .


*    light = '@5B@'.   "正确，绿灯
*    -light = '@5C@'.   "错误，红灯



      IF it_output-msg IS INITIAL.
        it_output-light = '@5B@'  .
        it_output-msg =  '库存地点扩展成功'.
      ENDIF.
      APPEND  it_output .
      CLEAR it_output.
      REFRESH it_bdc_tab[].
      REFRESH it_mesgtab[].

    ENDLOOP.
  ENDLOOP.
  PERFORM  frm_it_output .

  m_alv_fieldcat 'MSG'        '消息文本'            '40'.
  gs_layout-cwidth_opt = 'X'.
  g_repid = sy-repid.
  CALL FUNCTION 'REUSE_ALV_GRID_DISPLAY_LVC'

      is_layout_lvc      = gs_layout
      it_fieldcat_lvc    = gt_fieldcat
    TABLES
      t_outtab           = it_output[].

ENDFORM. " FRM_STK_OUTPUT

FORM frm_call_tcode USING p_tcode.

  CALL TRANSACTION p_tcode USING it_bdc_tab
                            MODE mode
                        MESSAGES INTO it_mesgtab.

ENDFORM. "frm_call_tcode
