FORM catlg_set USING p_field p_text.
  DATA: ls_fldct TYPE slis_fieldcat_alv.

  ls_fldct-fieldname     =  p_field.
  ls_fldct-seltext_l     =  p_text.

  CASE ls_fldct-fieldname.
    WHEN 'BSTMG' OR 'BDMNG' OR 'GAMNG'.
      ls_fldct-qfieldname = 'GMEIN'.
      ls_fldct-decimals_out = 3.
      ls_fldct-no_zero = 'X'.
    WHEN 'WRBTR' OR 'KBERE' OR 'BFAAC'.
      ls_fldct-cfieldname = 'BWAER'.
      ls_fldct-decimals_out = 2.

      ls_fldct-edit_mask = '==ALPHA'.
    WHEN 'MATNR' OR 'IDNRK'.

    WHEN 'GMEIN' OR 'MEINS' .

FORM status_f4 USING chkfld paraname.
  DATA: dynflds TYPE TABLE OF dynpread WITH HEADER LINE .

      request    = 'A'
    TABLES
      dynpfields = dynflds.
  DELETE dynflds WHERE  fieldname <> paraname.
  READ TABLE dynflds INDEX 1.
  CHECK sy-subrc = 0 .

  dynflds-fieldvalue = ` ` && dynflds-fieldvalue.
  LOOP AT it_tj02t.
    txt06 = ` ` && it_tj02t-txt04.

    IF dynflds-fieldvalue CS txt06.
      selectlist-selflag = 'X'.
    ENDIF.
    APPEND selectlist.CLEAR selectlist.
  ENDLOOP.
  CALL FUNCTION 'POPUP_TO_DECIDE_LIST'
    EXPORTING

      mark_max  = 0

FORM fillstab TABLES stattab USING chkflg stext.
  DATA it_stat TYPE TABLE OF jest-stat WITH HEADER LINE .

  CLEAR: stattab[].
  IF chkflg <> ''.
    IF stext = ''.
      MESSAGE s000(oo) WITH '状态选项如果不输请不要勾中'.
      STOP.
    ENDIF.
    SPLIT stext AT space INTO TABLE it_stat.
    LOOP AT it_stat.
      READ TABLE it_tj02t WITH KEY txt04 = it_stat.
      CONCATENATE 'IEQ' it_tj02t-istat INTO stattab.
      COLLECT stattab.
    ENDLOOP.
  ENDIF.
ENDFORM.
