AT SELECTION-SCREEN.





START-OF-SELECTION.
*权限检查
  REFRESH: it_tvak, r_auart.
  SELECT auart INTO TABLE it_tvak FROM tvak
    WHERE auart IN s_auart.
  LOOP AT it_tvak.
    AUTHORITY-CHECK OBJECT 'V_VBAK_AAT'
     ID 'AUART' FIELD it_tvak-auart
     ID 'ACTVT' FIELD '03'.
    IF sy-subrc = 0.
      r_auart+0(3) = 'IEQ'.
      r_auart-low  = it_tvak-auart.
      APPEND r_auart.
    ENDIF.
  ENDLOOP.
  IF r_auart[] IS INITIAL.
    MESSAGE i000 WITH '您没有查询订单类型的权限!' DISPLAY LIKE 'E'.
    EXIT.
  ENDIF.


*查询VBAK
  REFRESH it_vbak.
  SELECT * INTO TABLE it_vbak FROM vbak
    WHERE vbeln IN s_vbeln
      AND kunnr IN s_kunnr
      AND erdat IN s_erdat
      AND ernam IN s_ernam
      AND auart IN r_auart
      AND vkorg IN s_vkorg
      AND vtweg IN s_vtweg
      AND spart IN s_spart
      AND vkbur IN s_vkbur.
  IF it_vbak[] IS INITIAL.
    MESSAGE i000 WITH '查找不到任何数据!' DISPLAY LIKE 'E'.
    EXIT.
  ELSE.
*权限检查
    LOOP AT it_vbak.
      AUTHORITY-CHECK OBJECT 'V_VBAK_VKO'
       ID 'VKORG' FIELD it_vbak-vkorg
       ID 'VTWEG' FIELD it_vbak-vtweg
       ID 'SPART' FIELD it_vbak-spart
       ID 'ACTVT' FIELD '03'.
      IF sy-subrc <> 0.
        DELETE it_vbak.
      ENDIF.
    ENDLOOP.
    IF it_vbak[] IS INITIAL.
      MESSAGE i000 WITH '您没有查询销售组织/分销渠道/产品组的权限!' DISPLAY LIKE 'E'.
      EXIT.
    ENDIF.
  ENDIF.

*查询VBAP
  REFRESH it_vbap.
  SELECT * INTO TABLE it_vbap FROM vbap
    FOR ALL ENTRIES IN it_vbak
    WHERE vbeln =  it_vbak-vbeln
      AND matnr IN s_matnr.
  IF it_vbap[] IS INITIAL.
    MESSAGE i000 WITH '查找不到任何数据!' DISPLAY LIKE 'E'.
    EXIT.
  ENDIF.

*查询VBKD
  REFRESH it_vbkd.
  SELECT * INTO TABLE it_vbkd FROM vbkd
    FOR ALL ENTRIES IN it_vbak
    WHERE vbeln = it_vbak-vbeln
      AND posnr = ''.

*查询VBEP
  REFRESH it_vbep.
  SELECT * INTO TABLE it_vbep FROM vbep
    FOR ALL ENTRIES IN it_vbap
    WHERE vbeln = it_vbap-vbeln
      AND posnr = it_vbap-posnr.

*查询LIKP，LIPS
  REFRESH: it_likp, it_lips.
  SELECT * INTO TABLE it_lips FROM lips
    FOR ALL ENTRIES IN it_vbap
    WHERE vgbel = it_vbap-vbeln
      AND vgpos = it_vbap-posnr.
  IF NOT it_lips[] IS INITIAL.
    SELECT * INTO TABLE it_likp FROM likp
      FOR ALL ENTRIES IN it_lips
      WHERE vbeln = it_lips-vbeln.
  ENDIF.

*查询VBRK，VBRP
  REFRESH: it_vbrk, it_vbrp.
  SELECT * INTO TABLE it_vbrp FROM vbrp
    FOR ALL ENTRIES IN it_vbap
    WHERE vgbel = it_vbap-vbeln
      AND vgpos = it_vbap-posnr.
  IF NOT it_lips[] IS INITIAL.
    SELECT * APPENDING TABLE it_vbrp FROM vbrp
      FOR ALL ENTRIES IN it_lips
      WHERE vgbel = it_lips-vbeln
        AND vgpos = it_lips-posnr.
  ENDIF.
  IF NOT it_vbrp[] IS INITIAL.
    SELECT * INTO TABLE it_vbrk FROM vbrk
      FOR ALL ENTRIES IN it_vbrp
      WHERE vbeln =  it_vbrp-vbeln
        AND vbtyp IN ('M','O','P').
    LOOP AT it_vbrp.
      READ TABLE it_vbrk WITH KEY vbeln = it_vbrp-vbeln.
      IF sy-subrc <> 0.
        DELETE it_vbrp.
      ENDIF.
    ENDLOOP.
  ENDIF.

*查询VBFA：取发票冲销
  REFRESH it_vbfa.
  IF NOT it_vbrp[] IS INITIAL.
    SELECT * INTO TABLE it_vbfa FROM vbfa
      FOR ALL ENTRIES IN it_vbrp
      WHERE vbelv   =  it_vbrp-vbeln
        AND posnv   =  it_vbrp-posnr
        AND vbtyp_v IN ('M','O','P')
        AND vbtyp_n IN ('N','S').
  ENDIF.


*整理数据
*发票
  SORT it_vbrp BY vgbel vgpos vbeln posnr.
  REFRESH it_ykpsl.
  CLEAR: v_vgbel, v_vgpos.
  REFRESH it_show.
  LOOP AT it_vbrp.
    " 判断当前是否是最大的凭证
    v_tabix = sy-tabix.
    v_vgbel = it_vbrp-vgbel.
    v_vgpos = it_vbrp-vgpos.
    v_end   = ''.
    v_tabix = v_tabix + 1.
    READ TABLE it_vbrp INTO wa_vbrp INDEX v_tabix.
    IF sy-subrc = 0.
      IF wa_vbrp-vgbel <> v_vgbel OR wa_vbrp-vgpos <> v_vgpos.
        v_end = 'X'.
      ENDIF.
    ELSE.
      v_end = 'X'.
    ENDIF.
    CLEAR it_vbfa.
    READ TABLE it_vbfa WITH KEY vbelv = it_vbrp-vbeln
                                posnv = it_vbrp-posnr.
    IF sy-subrc = 0 AND v_end = ''. " 不是最大的一笔发票凭证且被冲销了不显示
      CONTINUE.
    ENDIF.

    " 匹配其他表
    CLEAR it_vbrk.
    READ TABLE it_vbrk WITH KEY vbeln = it_vbrp-vbeln.

    CLEAR: it_lips, it_likp, it_vbap, it_vbak, it_vbkd.
    READ TABLE it_lips WITH KEY vbeln = it_vbrp-vgbel
                                posnr = it_vbrp-posnr.
    IF sy-subrc = 0.
      READ TABLE it_likp WITH KEY vbeln = it_lips-vbeln.
      READ TABLE it_vbap WITH KEY vbeln = it_lips-vgbel
                                  posnr = it_lips-posnr.
    ELSE.
      READ TABLE it_vbap WITH KEY vbeln = it_vbrp-vgbel
                                  posnr = it_vbrp-posnr.
    ENDIF.
    READ TABLE it_vbak WITH KEY vbeln = it_vbap-vbeln.
    IF sy-subrc = 0.
      READ TABLE it_vbkd WITH KEY vbeln = it_vbak-vbeln.
    ENDIF.

    " 赋值到显示内表中
    CLEAR it_show.
    " 销售订单
    PERFORM sub_set_so.
    IF v_zexit = 'X'.
      CONTINUE.
    ENDIF.
    " 交货单
    PERFORM sub_set_dn.
    " 发票
    PERFORM sub_set_bl.

    APPEND it_show.
  ENDLOOP.

  " 计算未开票数量，未开票含税金额
  REFRESH: it_ykpsl_so, it_ykpsl_dn.
  it_ykpsl_so[] = it_ykpsl[].
  it_ykpsl_dn[] = it_ykpsl[].
  LOOP AT it_show.
    CLEAR wa_tvap.
    SELECT SINGLE * INTO wa_tvap FROM tvap
      WHERE pstyv = it_show-pstyv.
    CASE wa_tvap-fkrel.
      WHEN '' OR 'J'.
        it_show-zwkpsl = 0.
      WHEN 'A'.
        it_show-zwkpsl = it_show-lfimg.
        LOOP AT it_ykpsl_dn WHERE vbeln_dn = it_show-vbeln_dn
                              AND posnr_dn = it_show-posnr_dn.
          it_show-zwkpsl = it_show-zwkpsl - it_ykpsl_dn-fkimg.
        ENDLOOP.
        it_show-zwkphsje = it_show-zhsdj * it_show-zwkpsl.
      WHEN 'B' OR 'C'.
        it_show-zwkpsl = it_show-zddsl.
        LOOP AT it_ykpsl_so WHERE vbeln = it_show-vbeln
                              AND posnr = it_show-posnr.
          it_show-zwkpsl = it_show-zwkpsl - it_ykpsl_so-fkimg.
        ENDLOOP.
        it_show-zwkphsje = it_show-zhsdj * it_show-zwkpsl.
    ENDCASE.
    MODIFY it_show.
  ENDLOOP.

*交货单
  LOOP AT it_lips.
    READ TABLE it_show WITH KEY vbeln_dn = it_lips-vbeln
                                posnr_dn = it_lips-posnr.
    IF sy-subrc <> 0.
      READ TABLE it_likp WITH KEY vbeln = it_lips-vbeln.
      READ TABLE it_vbap WITH KEY vbeln = it_lips-vgbel
                                  posnr = it_lips-posnr.
      IF sy-subrc = 0.
        READ TABLE it_vbak WITH KEY vbeln = it_vbap-vbeln.
        IF sy-subrc = 0.
          READ TABLE it_vbkd WITH KEY vbeln = it_vbak-vbeln.
        ENDIF.
      ENDIF.

      " 赋值到显示内表中
      CLEAR it_show.
      " 销售订单
      PERFORM sub_set_so.
      IF v_zexit = 'X'.
        CONTINUE.
      ENDIF.
      " 交货单
      PERFORM sub_set_dn.

      APPEND it_show.
    ENDIF.
  ENDLOOP.

*销售订单
  LOOP AT it_vbap.
