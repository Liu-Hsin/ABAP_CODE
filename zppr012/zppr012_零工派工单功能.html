FORM handle_data_changed USING p_er_data_changed
      TYPE REF TO   cl_alv_changed_data_protocol.

  FIELD-SYMBOLS <fs_out> LIKE LINE OF gt_out.
  DATA:x_change TYPE lvc_s_modi .
  CLEAR:gt_errormsg.

  LOOP AT p_er_data_changed->mt_good_cells INTO x_change.

    CASE x_change-fieldname.
      WHEN 'CTGRY'. "类别
        READ TABLE gt_out ASSIGNING <fs_out> INDEX x_change-row_id.
        IF sy-subrc = 0.
          PERFORM get_dd07t USING 'ZDECTG'  x_change-value CHANGING <fs_out>-ctgryt.
        ENDIF.

        IF x_change-value IS NOT INITIAL.

          CLEAR:GS_zppT001.
          READ TABLE GT_zppT001 INTO GS_zppT001 WITH KEY ctgry = x_change-value.

          IF sy-subrc <> 0.
            gs_errormsg-msgty = 'E'.
            gs_errormsg-arbgb = 'Z001'.
            gs_errormsg-txtnr =  '001'.
            gs_errormsg-msgv1 = |行：{ x_change-row_id }：类别没有权限|.
            APPEND gs_errormsg TO gt_errormsg..
          ENDIF.

        ENDIF.

      WHEN 'WRKSHP'. "车间
